005import java.util.List;
006import java.util.Set;
007import java.util.stream.Collectors;
008
009import org.jgrapht.Graph;
010import org.jgrapht.UndirectedGraph;
011import org.jgrapht.alg.HamiltonianCycle;
012import org.jgrapht.alg.cycle.HierholzerEulerianCycle;
013import org.jgrapht.alg.cycle.PatonCycleBase;
014import org.jgrapht.alg.cycle.UndirectedCycleBase;
015import org.jgrapht.graph.GraphWalk;
016import org.jgrapht.graph.SimpleWeightedGraph;
017
018import us.lsi.graphs.GraphWalk2;
019import us.lsi.graphs.Graphs2;
020import us.lsi.graphs.GraphsReader;
021
022public class Ciclos {
023
024        public static void main(String[] args) {
025                
026                Graph<Ciudad,Carretera> graph =  new SimpleWeightedGraph<Ciudad,Carretera>(Carretera::create);
027                graph =  GraphsReader.newGraph("./ficheros/andalucia.txt",Ciudad::create, Carretera::create,graph,Carretera::getKm);
028                
029                SimpleWeightedGraph<Ciudad, Carretera> gc = Graphs2.completeGraph(graph,200000.);
030                
031                
032                List<Ciudad> r3 = HamiltonianCycle.getApproximateOptimalForCompleteGraph(gc);
033                GraphWalk<Ciudad,Carretera> gw = new GraphWalk<>(gc,r3,0.);
034                
035                System.out.println("Hamiltonian");
036                System.out.println(r3);
037                System.out.println(gw.getEdgeList());
038                
039                
040                
041                final Graph<Ciudad,Carretera> g = graph;
042                
043                HierholzerEulerianCycle<Ciudad,Carretera> hc = new HierholzerEulerianCycle<>();
044                Boolean r2 = hc.isEulerian(graph);
045                
046                System.out.println(r2);
047                
048                UndirectedCycleBase<Ciudad,Carretera> sc = new PatonCycleBase<Ciudad,Carretera>((UndirectedGraph<Ciudad, Carretera>) graph);
049                List<List<Ciudad>> r = sc.findCycleBase();
050                
051                Set<GraphWalk2<Ciudad,Carretera>> sgw = 
052                                r.stream().map(x->new GraphWalk2<Ciudad,Carretera>(g,x))
053                                                .collect(Collectors.toSet());
054                
055                sgw.stream().forEach(x->System.out.println(x.getWeight()+"==="+x.getLength()+"==="+x.getVertexList()));
056        }
057
058}
