001package us.lsi.graphs;
002
003import java.util.List;
004
005import org.jgrapht.Graph;
006import org.jgrapht.GraphPath;
007import org.jgrapht.graph.GraphWalk;
008
009import com.google.common.base.Preconditions;
010import com.google.common.collect.Lists;
011
012/**
013 * @author Miguel Toro
014 * 
015 * <a> Clase de factoría para el tipo GraphPath&lt;V,E&gt; </a>
016 *
017 */
018public class GraphPaths {
019
020        public static <V,E> GraphPath<V,E> addGraphPath(GraphPath<V,E> r1, GraphPath<V,E> r2){
021                Preconditions.checkArgument(r1.getEndVertex().equals(r2.getStartVertex()),"El vértice final no es igual al inicial");
022                List<E> edgeList = Lists.newArrayList(r1.getEdgeList());
023                edgeList.addAll(r2.getEdgeList());
024                return new GraphWalk<V,E>(r1.getGraph(), r1.getStartVertex(), r2.getEndVertex(),
025                                edgeList, r1.getWeight()+r2.getWeight());
026        }
027
028        public static <V,E> GraphPath<V,E> create(Graph<V,E> g, V v1, V v2, E e, Double d){
029                List<E> edgeList = Lists.newArrayList();
030                edgeList.add(e);
031                return new GraphWalk<V,E>(g, v1, v2, edgeList, d);
032        }
033}
