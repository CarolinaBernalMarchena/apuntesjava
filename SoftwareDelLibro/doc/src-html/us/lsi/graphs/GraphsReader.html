001package us.lsi.graphs;
002
003import java.io.File;
004import java.io.FileNotFoundException;
005import java.util.ArrayList;
006import java.util.HashMap;
007import java.util.List;
008import java.util.Map;
009import java.util.Scanner;
010
011import org.jgrapht.Graph;
012import org.jgrapht.WeightedGraph;
013
014/**
015 * <p>
016 * Clase adecuada para construir un grafo a partir de la información en un fichero
017 * 
018 * El fichero debe tener una estructura como la siguiente. 
019 * 
020 * Cada línea en la sección de vértices representa un vértice. 
021 * El primer campo debe ser un identificador único para el mismo
022 * Los otros campos propiedades
023 * 
024 * Cada línea de la sección de aristas representa una arista
025 * Los dos primeros campos son los identificadores de los vértices extremos
026 * Los otros campos propiedades
027 * </p>
028 * 
029 * #VERTEX# <br>
030 * s1,Sevilla <br>
031 * s2,Córdoba <br>
032 * s3,Cádiz <br>
033 * s4,Málaga <br>
034 * #EDGE# <br>
035 * s1,s2,180. <br>
036 * s1,s3,120. <br>
037 * s2,s4,140. <br>
038 * s4,s1,210. <br>
039 *
040 * 
041 * 
042 * @author Miguel Toro
043 *
044 */
045public class GraphsReader {
046
047        public static String[] eliminaBlancos(String[] s) {
048                String[] r = new String[s.length];
049                for (int i = 0; i < r.length; i++) {
050                        r[i] = s[i].trim();
051                }
052                return r;
053        }
054
055        /**
056         * @param file Fichero con la información de entrada
057         * @param vf Factoría para construir los vértices a partir de las líneas del fichero previamente tokenizada
058         * @param ef Factoría para construir las aristas a partir de las líneas del fichero previamente tokenizada
059         * @param graph Grafo de entrada 
060         * @param ew Función que proporciona el peso de la arista. Debe ser distinto de null si el grafo es 
061         * de tipo WeightedGraph&lt;V,E&gt;      
062         * @return Grafo con la información del fichero
063         * @param <V> El tipo de los vértices
064         * @param <E> El tipo de las aristas
065         * @throws IllegalArgumentException Si no encuentra el fichero o si el fichero tiene una estructura no adecuada
066         */
067        public static <V, E> Graph<V, E> newGraph(String file,
068                        StringVertexFactory<V> vf, StringEdgeFactory<V, E> ef,
069                        Graph<V, E> graph,
070                        EdgeWeight<E> ew) {
071
072                Map<String, V> idVertices = new HashMap<>();
073                Graph<V, E> ret = graph;
074                WeightedGraph<V,E> wg = null;
075                if (ew!=null) wg = (WeightedGraph<V, E>) graph;
076                Scanner sc = null;
077                try {
078                        sc = new Scanner(new File(file));
079                } catch (FileNotFoundException e1) {
080                        throw new IllegalArgumentException("Fichero " + file + " no encontrado");
081                }
082
083                List<String> filas = new ArrayList<>();
084
085                while (sc.hasNext()) {
086                        filas.add(sc.nextLine());
087                }
088
089                sc.close();
090                
091                try {
092                        List<String> vertices = filas.subList(1, filas.indexOf("#EDGE#"));
093                        List<String> aristas = filas.subList(filas.indexOf("#EDGE#") + 1, filas.size());
094
095                        for (String verticeStr : vertices) {
096                                String[] vertice = eliminaBlancos(verticeStr.split(","));
097
098                                if (vertice.length == 0) {
099                                        throw new IllegalArgumentException(
100                                                        "El fichero de entrada de vertices no es correcto. Linea: "
101                                                                        + verticeStr);
102                                }
103                                V vertex = vf.createVertex(vertice);
104                                ret.addVertex(vertex);
105                                idVertices.put(vertice[0], vertex);
106                        }
107                        
108                        for (String aristaStr : aristas) {
109                                String[] arista = eliminaBlancos(aristaStr.split(","));
110                                
111                                if (arista.length < 2)
112                                        throw new IllegalArgumentException("El número de vértices de la arista no es correcto");
113
114                                if (arista.length == 2) {
115                                        ret.addEdge(idVertices.get(arista[0]),
116                                                        idVertices.get(arista[1]));
117                                } else if (arista.length > 2) {
118                                        E edge = ef.createEdge(idVertices.get(arista[0]),
119                                                        idVertices.get(arista[1]), arista);
120                                        ret.addEdge(idVertices.get(arista[0]),
121                                                        idVertices.get(arista[1]), edge);
122                                        if(wg!=null){
123                                                wg.setEdgeWeight(edge, ew.getWeight(edge));
124                                        }
125                                }
126                        }
127
128                        return ret;
129
130                } catch (IndexOutOfBoundsException e) {
131                        throw new IllegalArgumentException(
132                                        "La composición del fichero no es correcta");
133                }
134        }
135}
