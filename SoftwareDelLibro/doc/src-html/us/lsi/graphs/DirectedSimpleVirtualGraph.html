001package us.lsi.graphs;
002
003import java.util.HashSet;
004import java.util.Set;
005import org.jgrapht.Graph;
006/**
007 * <p> Implementación de un grafo virtual simple y dirigido
008 * Asumimos que los vértices son subtipo de VirtualDirectedVertex &lt; V,E &gt;
009 * Asumimos que las aristas son subtipos de SimpleEdge &lt; V &gt; 
010 * </p>
011 * 
012 * @see us.lsi.graphs.VirtualDirectedVertex
013 * 
014 * <p> El grafo es no modificable por lo que no están permitadas las operación de modificación </p>
015 * 
016 * @author Miguel Toro
017 *
018 * @param <V> El tipo de los vértices
019 * @param <E> El tipo de las aristas
020 * 
021 */
022
023public class DirectedSimpleVirtualGraph<V extends VirtualDirectedVertex<V,E>, E extends SimpleEdge<V>>
024                extends SimpleVirtualGraph<V, E> 
025                implements Graph<V, E> {
026                
027
028        public DirectedSimpleVirtualGraph() {
029                super();
030        }
031                
032        @SafeVarargs
033        public DirectedSimpleVirtualGraph(V... vertexSet) {
034                super(vertexSet);
035        }
036
037
038
039        @Override
040        public Set<E> edgesOf(V v) {
041                Set<E> r = new HashSet<>(this.incomingEdgesOf(v));
042                r.addAll(this.outgoingEdgesOf(v));
043                return r;
044        }
045        
046        @Override
047        public int inDegreeOf(V v) {
048                return v.incomingEdgesOf().size();
049        }
050
051        @Override
052        public Set<E> incomingEdgesOf(V v) {
053                return v.incomingEdgesOf();
054        }
055
056        @Override
057        public int outDegreeOf(V v) {
058                return v.outgoingEdgesOf().size();
059        }
060
061        @Override
062        public Set<E> outgoingEdgesOf(V v) {
063                return v.outgoingEdgesOf();
064        }
065        
066}
