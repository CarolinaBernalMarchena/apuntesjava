001package us.lsi.graphs;
002
003
004/**
005 * @author Miguel Toro
006 *
007 * @param <V> Tipo de los vértices que conecta la arista
008 * @param <A> Tipo de la acción asociada a la arista
009 */
010public class ActionSimpleEdge<A, V> extends SimpleEdge<V> {
011
012        /**
013         * 
014         */
015        private static final long serialVersionUID = 1L;
016
017        public static <A,V> ActionSimpleEdge<A,V> create(V c1, V c2, A a) {
018                return new ActionSimpleEdge<A,V>(c1, c2, a);
019        }
020
021
022        public static <A,V> ActionSimpleEdge<A,V> create(V c1, V c2, A a,
023                        double weight) {
024                return new ActionSimpleEdge<A,V>(c1, c2, a, weight);
025        }
026
027        private A accion;
028        
029        protected ActionSimpleEdge(V c1, V c2, A a) {
030                super(c1, c2);
031                this.accion = a;
032        }
033        
034
035        protected ActionSimpleEdge(V c1, V c2, A a, double weight) {
036                super(c1, c2, weight);
037                this.accion = a;
038        }
039
040        public A getAction() {
041                return accion;
042        }
043
044
045        @Override
046        public int hashCode() {
047                final int prime = 31;
048                int result = super.hashCode();
049                result = prime * result
050                                + ((accion == null) ? 0 : accion.hashCode());
051                return result;
052        }
053
054
055        @Override
056        public boolean equals(Object obj) {
057                if (this == obj)
058                        return true;
059                if (!super.equals(obj))
060                        return false;
061                if (!(obj instanceof ActionSimpleEdge))
062                        return false;
063                ActionSimpleEdge<?,?> other = (ActionSimpleEdge<?,?>) obj;
064                if (accion == null) {
065                        if (other.accion != null)
066                                return false;
067                } else if (!accion.equals(other.accion))
068                        return false;
069                return true;
070        }
071
072
073        @Override
074        public String toString() {
075                return super.toString()+","+this.accion;
076        }
077
078        
079        
080
081}
