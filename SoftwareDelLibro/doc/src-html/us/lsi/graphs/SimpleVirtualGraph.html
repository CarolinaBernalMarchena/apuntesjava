001package us.lsi.graphs;
002
003import java.util.*;
004
005import org.jgrapht.*;
006
007/**
008 * <p> Implementación de un grafo virtual simple 
009 * Asumimos que los vértices son subtipo de VirtualVertex &lt; V,E &gt;
010 * Asumimos que las aristas son subtipos de SimpleEdge &lt; V &gt; 
011 * </p>
012 * 
013 * <p> El grafo es inmutable por lo que no están permitadas las operación de modificación. Tampoco
014 * están permitidas las operaciones de consulta de todos los vértices o todas las aristas.
015 *  Si se invocan alguna de ellas se disparará 
016 * la excepción UnsupportedOperationException </p>
017 * 
018 * @see us.lsi.graphs.VirtualVertex
019 * 
020 * 
021 * 
022 * @author Miguel Toro
023 *
024 * @param <V> El tipo de los vértices
025 * @param <E> El tipo de las aristas
026 * 
027 */
028/**
029 * @author Boss
030 *
031 * @param <V> Tipo de los vértices
032 * @param <E> Tipo de las aristas
033 */
034public class SimpleVirtualGraph<V extends VirtualVertex<V,E>, E extends SimpleEdge<V>>
035        implements Graph<V, E> {
036                        
037        private Set<V> vertexSet;
038        
039        public SimpleVirtualGraph(){
040                super();
041                this.vertexSet = new HashSet<>();
042        }
043        
044        /**
045         * @param vertexSet Conjunto de vértices del grafo que queremos hacer explícitos.
046         */
047        @SafeVarargs
048        public SimpleVirtualGraph(V... vertexSet){
049                super();
050                this.vertexSet = new HashSet<>();
051                for(V v:vertexSet){
052                        this.vertexSet.add(v);
053                }
054        }
055        
056        @Override
057        public EdgeFactory<V, E> getEdgeFactory() {
058                return null;
059        }
060
061        @Override
062        public boolean containsEdge(E e) {
063                return e.getSource().isNeighbor(e.getTarget());
064        }
065
066        @Override
067        public boolean containsEdge(V v1, V v2) {
068                return v1.isNeighbor(v2);
069        }
070        @Override
071        public boolean containsVertex(V v) {
072                return v.isValid();
073        }
074        @Override
075        public V getEdgeSource(E e) {
076                return e.getSource();
077        }
078
079        @Override
080        public V getEdgeTarget(E e) {
081                return e.getTarget();
082        }
083
084        @Override
085        public double getEdgeWeight(E e) {
086                return e.getEdgeWeight();
087        }
088        
089        @Override
090        public E getEdge(V v1, V v2) {
091                E a = null;
092                if (v1.isNeighbor(v2)) {
093                        a = v1.edgesOf()
094                                        .stream()
095                                        .filter(e->e.otherVertex(v1).equals(v2))
096                                        .findFirst()
097                                        .get();
098                }
099                return a;
100        }
101        
102        @Override
103        public Set<E> getAllEdges(V v1, V v2) {
104                Set<E> s = new HashSet<>();
105                if (v1.isNeighbor(v2))
106                        s.add(getEdge(v1, v2));
107                return s;
108        }
109        
110        
111        
112        /** 
113         * @return Conjunto de vértices del grafo que se han hecho explícitos en el constructor.
114         */
115        @Override
116        public Set<V> vertexSet(){
117                return this.vertexSet;
118        }
119        
120        @Override
121        public Set<E> edgesOf(V v) {
122                return v.edgesOf();
123        }
124        
125        
126        /**
127         * @see org.jgrapht.Graph#edgeSet()
128         * @throw UnsupportedOperationException
129         */
130        @Override
131        public Set<E> edgeSet() {
132                throw new UnsupportedOperationException();
133        }
134        
135        /**
136         * @see org.jgrapht.Graph#addEdge(java.lang.Object, java.lang.Object)
137         * @throw UnsupportedOperationException
138         */
139        @Override
140        public E addEdge(V arg0, V arg1) {
141                throw new UnsupportedOperationException();
142        }
143
144        
145        /**
146         * @see org.jgrapht.Graph#addEdge(java.lang.Object, java.lang.Object, java.lang.Object)
147         * @throw UnsupportedOperationException
148         */
149        @Override
150        public boolean addEdge(V arg0, V arg1, E arg2) {
151                throw new UnsupportedOperationException();
152        }
153
154
155        /**
156         * @see org.jgrapht.Graph#addVertex(java.lang.Object)
157         * @throw UnsupportedOperationException
158         */
159        @Override
160        public boolean addVertex(V arg0) {
161                throw new UnsupportedOperationException();
162        }
163
164        /**
165         * @see org.jgrapht.Graph#removeAllEdges(java.util.Collection)
166         * @throw UnsupportedOperationException
167         */
168        @Override
169        public boolean removeAllEdges(Collection<? extends E> arg0) {
170                throw new UnsupportedOperationException();
171        }
172
173        /**
174         * 
175         * @throw UnsupportedOperationException
176         */
177        @Override
178        public Set<E> removeAllEdges(V arg0, V arg1) {
179                throw new UnsupportedOperationException();
180        }
181
182        /**
183         * @see org.jgrapht.Graph#removeAllVertices(java.util.Collection)
184         * @throw UnsupportedOperationException
185         */
186        @Override
187        public boolean removeAllVertices(Collection<? extends V> arg0) {
188                throw new UnsupportedOperationException();
189        }
190
191        /**
192         * @see org.jgrapht.Graph#removeEdge(java.lang.Object)
193         * @throw UnsupportedOperationException
194         */
195        @Override
196        public boolean removeEdge(E arg0) {
197                throw new UnsupportedOperationException();
198        }
199
200        /**
201         * @see org.jgrapht.Graph#removeEdge(java.lang.Object, java.lang.Object)
202         * @throw UnsupportedOperationException
203         */
204        @Override
205        public E removeEdge(V arg0, V arg1) {
206                throw new UnsupportedOperationException();
207        }
208
209        /**
210         * @see org.jgrapht.Graph#removeVertex(java.lang.Object)
211         * @throw UnsupportedOperationException
212         */
213        @Override
214        public boolean removeVertex(V arg0) {
215                throw new UnsupportedOperationException();
216        }       
217
218}
