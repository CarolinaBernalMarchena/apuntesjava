001package us.lsi.tiposrecursivos;
002
003import java.util.Map;
004
005import us.lsi.algoritmos.AbstractAlgoritmo;
006import us.lsi.common.Maps2;
007
008/**
009 * @author Miguel Toro
010 * 
011 * <p> Expresión binaria
012 *
013 * @param <R> Tipo del resultado 
014 * @param <T1> Tipo del primer operando
015 * @param <T2> Tipo del segundo operando
016 */
017public class BinaryExp<T1,T2,R> extends Exp<R> {
018
019        private Exp<T1> op1 = null;
020        private Exp<T2> op2 = null;
021        private BinaryOperatorExp<T1,T2,R> operator = null;
022        private int id;
023        private static int lastId = 0;
024        private Boolean flag = false;
025        
026        BinaryExp(Exp<T1> left, Exp<T2> right, BinaryOperatorExp<T1,T2,R> operator) {
027                super();
028                this.op1 = left;
029                this.op2 = right;
030                this.operator = operator;
031                this.id = lastId;
032                lastId++;
033        }
034        
035        @Override
036        public Integer getArity() {
037                return 2;
038        }
039
040        public Exp<T1> getOp1() {
041                return op1;
042        }
043
044        public void setOp1(Exp<T1> op1) {
045                this.op1 = op1;
046        }
047
048        public Exp<T2> getOp2() {
049                return op2;
050        }
051
052        public void setOp2(Exp<T2> op2) {
053                this.op2 = op2;
054        }
055
056        public void setOp(Exp<T1> op1, Exp<T2> op2) {
057                this.op1 = op1;
058                this.op2 = op2;
059        }
060
061        public BinaryOperatorExp<T1, T2, R> getOperator() {
062                return operator;
063        }
064
065        @Override
066        public String getName() {
067                return operator.getName();
068        }
069        
070        @Override
071        public R val() {
072                return this.operator.getCode().apply(op1.val(), op2.val());
073        }
074        
075        public String toString() {      
076                String sOp1 = this.getOp1().toString();
077                String sOp2 = this.getOp2().toString();
078                if(op1.getPriority() < this.getPriority()){
079                        sOp1 = "("+sOp1+")";
080                }
081                if(op2.getPriority() < this.getPriority()){
082                        sOp2 = "("+sOp2+")";
083                }
084                return String.format(sOp1+operator.getShortName()+sOp2);
085        }
086
087        @Override
088        public Exp<R> copy() {
089                return Exp.binary(this.op1.copy(), this.op2.copy(), this.operator);
090        }
091
092        public Boolean isBinary() {
093                return true;
094        }
095        
096        @Override
097        public Boolean isConstant() {
098                return this.op1.isConstant() && this.op2.isConstant();
099        }
100
101        @Override
102        public Exp<R> simplify() {
103                Exp<R> r;
104                if(op1.isConstant() && op2.isConstant()){
105                        r = Exp.constant(this.val());
106                } else {
107                        r = Exp.binary(op1.simplify(),op2.simplify(), operator);
108                } 
109                return r;
110        }
111
112        @Override
113        public int hashCode() {
114                final int prime = 31;
115                int result = 1;
116                result = prime * result + ((op1 == null) ? 0 : op1.hashCode());
117                result = prime * result + ((op2 == null) ? 0 : op2.hashCode());
118                result = prime * result
119                                + ((operator == null) ? 0 : operator.hashCode());
120                return result;
121        }
122
123        @Override
124        public boolean equals(Object obj) {
125                if (this == obj)
126                        return true;
127                if (obj == null)
128                        return false;
129                if (!(obj instanceof BinaryExp))
130                        return false;
131                BinaryExp<?,?,?> other = (BinaryExp<?,?,?>) obj;
132                if (op1 == null) {
133                        if (other.op1 != null)
134                                return false;
135                } else if (!op1.equals(other.op1))
136                        return false;
137                if (op2 == null) {
138                        if (other.op2 != null)
139                                return false;
140                } else if (!op2.equals(other.op2))
141                        return false;
142                if (operator == null) {
143                        if (other.operator != null)
144                                return false;
145                } else if (!operator.equals(other.operator))
146                        return false;
147                return true;
148        }
149
150        @Override
151        public Boolean match(Exp<?> exp) {
152                Boolean r = false;
153                if(exp.isPlaceHolderExp()){
154                        exp.<R>asPlaceHolderExp().setExp(this);
155                        r = true;
156                } else if(exp.isBinary() && this.operator.equals(exp.getOperator())){
157                        BinaryExp<?,?,?> t = exp.asBinary();
158                        r = this.op1.match(t.op1) && this.op2.match(t.op2);
159                }
160                return r;
161        }
162
163        @Override
164        protected String getId() {
165                return "BE_"+id;
166        }
167
168        @Override
169        protected void toDOT(String file) {
170                if (!flag) {
171                        String s1 = "    \"%s\" [label=\"%s\"];";
172                        AbstractAlgoritmo.getFile().println(
173                                        String.format(s1, getId(), this.operator.getShortName()));
174                        String s2 = "    \"%s\" -> \"%s\";";
175                        AbstractAlgoritmo.getFile().println(
176                                        String.format(s2, getId(), op1.getId()));
177                        AbstractAlgoritmo.getFile().println(
178                                        String.format(s2, getId(), op2.getId()));
179                        op1.toDOT(file);
180                        op2.toDOT(file);
181                }
182                flag = true;
183        }
184        
185        @Override
186        protected void setFlagFalse() {
187                flag = false;
188                op1.setFlagFalse();
189                op2.setFlagFalse();
190        }
191        
192        @Override
193        public ExpType getExpType() {
194                return operator.getExpType()[2];
195        }
196        
197        @Override
198        protected Map<String, Exp<?>> vars() {
199                Map<String, Exp<?>> r = Maps2.newHashMap(op1.vars());
200                r.putAll(op2.vars());;
201                return r;
202        }
203        
204        @Override
205        public Integer getPriority() {
206                return this.operator.getPriority();
207        }
208        
209        @Override
210        public Exp<R> ifMatchTransform(Exp<?> pattern, Map<String,Exp<?>> vars, String patternResult) {
211                Exp<T1> r1 = op1.ifMatchTransform(pattern, vars, patternResult);
212                Exp<T2> r2 = op2.ifMatchTransform(pattern,vars, patternResult);
213                Exp<R> r = Exp.binary(r1,r2, this.operator);
214                Exp<?> copy = pattern.copy();
215                if(r.match(copy)){
216                        Map<String,Exp<?>> m = copy.getVars();
217                        Map<String,Exp<?>> m2 = 
218                                        Maps2.<String,Exp<?>,Exp<?>>newHashMap(m, 
219                                                        x->x.isPlaceHolderExp()?x.asPlaceHolderExp().getExp():x);                       
220                        m2.putAll(vars);
221                        r = Exp.string(patternResult,m2);
222                }
223                return r;
224        }
225}
