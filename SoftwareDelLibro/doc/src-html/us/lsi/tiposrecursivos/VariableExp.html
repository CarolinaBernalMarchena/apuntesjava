001package us.lsi.tiposrecursivos;
002
003import java.util.Map;
004
005import us.lsi.algoritmos.AbstractAlgoritmo;
006import us.lsi.common.Maps2;
007import us.lsi.common.Preconditions;
008
009public class VariableExp<R> extends Exp<R> implements Operator {
010
011        private R value;
012        private String name;
013        private int id;
014        private static int lastId = 0;
015        private Boolean flag = false;
016        ExpType expType;        
017        
018        VariableExp(String name, R value, ExpType expType) {
019                super();
020                this.name = name;
021                this.value = value;
022                this.expType = expType;
023                this.id = lastId;
024                lastId++;
025        }
026
027        @Override
028        public Integer getArity() {
029                return 0;
030        }
031
032        @Override
033        public Operator getOperator() {
034                return this;
035        }
036        
037        @Override
038        public R val() {
039                Preconditions.checkState(value!=null,String.format("La variable %s tiene valor null",name));
040                return value;
041        }
042
043        public R getValue() {
044                return value;
045        }
046
047        public void setValue(R value) {
048                this.value = value;
049        }
050        
051        public String getId() {
052                return "V_"+id;
053        }
054
055        @Override
056        public String toString() {
057                return name;
058        }
059
060        @Override
061        public Exp<R> copy() {
062                return Exp.variable(this.name, this.value, this.expType);
063        }
064
065
066        public String getName() {
067                return name;
068        }
069        
070        public Boolean isVariable() {
071                return true;
072        }
073
074        @Override
075        public Boolean match(Exp<?> exp) {
076                Boolean r = false;
077                if(exp.isPlaceHolderExp()){
078                        exp.<R>asPlaceHolderExp().setExp(this);
079                        r = true;
080                } else if(exp.isVariable()){
081                        if(this.getName().equals(exp.asVariable().getName())) r = true;
082                }
083                return r;
084        }
085        
086        @Override
087        public Boolean isConstant() {
088                return false;
089        }
090
091        @Override
092        public Exp<R> simplify() {
093                return this;
094        }
095
096        @Override
097        public int hashCode() {
098                final int prime = 31;
099                int result = 1;
100                result = prime * result + ((name == null) ? 0 : name.hashCode());
101                return result;
102        }
103
104        @Override
105        public boolean equals(Object obj) {
106                if (this == obj)
107                        return true;
108                if (obj == null)
109                        return false;
110                if (!(obj instanceof VariableExp))
111                        return false;
112                VariableExp<?> other = (VariableExp<?>) obj;
113                if (name == null) {
114                        if (other.name != null)
115                                return false;
116                } else if (!name.equals(other.name))
117                        return false;
118                return true;
119        }
120
121        @Override
122        protected void toDOT(String file) {
123                if (!flag) {
124                        String s = "    \"%s\" [label=\"%s\"];";
125                        AbstractAlgoritmo.getFile().println(
126                                        String.format(s, getId(), getShortName()));
127                }       
128                flag = true;
129        }
130
131        @Override
132        public String getShortName() {
133                return getName();
134        }
135        
136        @Override
137        protected void setFlagFalse() {
138                flag = false;
139        }
140        
141        @Override
142        public ExpType getExpType() {
143                return expType;
144        }
145
146        @Override
147        protected Map<String, Exp<?>> vars() {
148                return Maps2.newHashMap(name, this);
149        }
150
151        @Override
152        public Integer getPriority() {
153                return 12;
154        }
155
156        @Override
157        public Exp<R> ifMatchTransform(Exp<?> pattern,Map<String,Exp<?>> vars,String result) {
158                return this;
159        }
160        
161}
