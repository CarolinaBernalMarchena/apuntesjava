183                Preconditions.checkArgument(c2.compareTo(c1) > 0);
184                Stream2.fromFile(fileIn)

200        public enum Cuadrante{PRIMER_CUADRANTE, SEGUNDO_CUADRANTE, TERCER_CUADRANTE, CUARTO_CUADRANTE};
201        
202        public static Cuadrante getCuadrante(Punto2D p){
203                Cuadrante c;
204                if(p.getX() >=0 && p.getY() >=0){
205                        c = Cuadrante.PRIMER_CUADRANTE;
206                } else if(p.getX() <=0 && p.getY() >=0){
207                        c = Cuadrante.SEGUNDO_CUADRANTE;
208                } else if(p.getX() <=0 && p.getY() <=0){
209                        c = Cuadrante.TERCER_CUADRANTE;
210                } else {
211                        c = Cuadrante.CUARTO_CUADRANTE;
212                }
213                return c;
214        }

288         * @param n Un entero
289         * @return Si es primo
290         */
291        public static boolean esPrimo1(Long n){
292                Long sqrt = (long)Math.sqrt((double)n);
293                return !LongStream.rangeClosed(2, sqrt).anyMatch(x->Math2.esDivisible(n, x));
294        }
295        
296        /**
297         * @param n Un entero
298         * @return Si es primo
299         */
300        public static boolean esPrimo2(Long n){
301                Long sqrt = (long)Math.sqrt((double)n);
302                Long e = 2L;
303                Boolean a = false;
304                while(e <= sqrt){          
305                a = Math2.esDivisible(n, e);
306                if(a) break; 
307                        e = e + 1;                 
308        }
309                return !a;
310        }
311        /**
312         * @param n Un entero
313         * @return Primer primo mayor que a
314         */
315        public static Long siguientePrimo1(Long n){
316                Long e0 = n%2==0?n+1:n+2;
317                return Stream.iterate(e0, e->e+2).filter(e->Math2.esPrimo(e)).findFirst().get();
318        }
319        /**
320         * @param n Un entero
321         * @return Primer primo mayor que a
322         */
323        public static Long siguientePrimo2(Long n){
324                Long e = n%2==0?n+1:n+2;
325                Long a = null;
326                while(true){       
327                if(esPrimo2(e))  {
328                        a = e;
329                        break;
330                }
331                        e = e +2;                  
332        }
333                return a;
334        }
