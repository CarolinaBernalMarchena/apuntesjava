016        /**
017         * @param n Un entero
018         * @param k Un entero
019         * @return Valor del número combinatorio n sobre k calculado de forma iterativa
020         */
021        public static int binom(int n, int k) {
022                List<Integer> lsa = Arrays.asList(1);
023                int i = 1;
024                while (i <= n) {
025                        List<Integer> ls = Lists.newArrayList();
026                        for (int s = 0; s <= i; s++) {
027                                if (s == 0 || s == i) {
028                                        ls.add(1);
029                                } else if (s == 1 || s == i - 1) {
030                                        ls.add(i);
031                                } else {
032                                        ls.add(lsa.get(s - 1) + lsa.get(s));
033                                }
034                        }
035                        i = i + 1;
036                        lsa = Lists.newArrayList(ls);
037                }
038                return lsa.get(k);
039        }

041        /**
042         * @param b Base 
043         * @param n Exponente
044         * @return Resultado de elevar b a n calculado de forma iterativa
045         */
046        public static long pot(int b,int n){
047                int e = b;
048                int a = 1;
049                while( n > 0){
050                        if(n%2==1){
051                             a = a * e;
052                        }
053                        e = e * e;
054                        n = n/2;
055                }
056                return a;
057        }

059        /**
060         * Adaptación del algoritmo anterior pot(b,n) para el cálculo de los número de Fibonacci
061         * 
062         * @param n Término
063         * @return Valor del n-esimo número de Fibonacci calculado de forma iterativa
064         */
065        public static long fib(int n){
066                        int i = 0;
067                    int a = 1;
068                        int b = 0;
069                        while(i < n){
070                                i = i+1;
071                                int a0 = a;
072                                a = a0+b;
073                                b = a0;
074                        }
075                        return b;
076        }

079        /**
080         * Adaptación del algoritmo anterior pot(b,n), siendo b, a, e de tipo Matriz. Alternativamente 
081         * se podria usar el método e.power(n) para calcular la potencia de la basea asociada a la recurrencia
082         * 
083         * @param n Término de la secuencia
084         * @param cf Coeficientes de la secuencia
085         * @param vi Valores iniciales de la secuencia
086         * @return Valor del término n-esimo calculado de forma iterativa
087         */
088        public static BigInteger secuencia(Integer n, Integer[] cf, Integer[] vi){
089                if(cf.length != vi.length) return null;
090                int k = cf.length;
091                FieldMatrix<BigFraction> e = Matrices.getBase(k,cf);                    
092                FieldMatrix<BigFraction> a = Matrices.getId(k,BigFractionField.getInstance());
093                FieldMatrix<BigFraction> c = Matrices.getColumn(vi);
094                while(n > 0){
095                if(n%2==1){
096                             a = a.multiply(e);
097                        }
098                        e = e.multiply(e);
099                        n = n/2;
100                }       
101                FieldMatrix<BigFraction> r = a.multiply(c);
102                return r.getEntry(1, 0).getNumerator();
103}
104
105        /**
106         * @param a Un entero
107         * @param b Un entero
108         * @return Máximo Común Divisor de los dos enteros calculado de forma iterativa
109         */
110        public static long mcd(int a,int b) {
111                while(b > 0){
112                   int a0 = a;
113                   a = b;
114                   b = a0%b;   
115            }
116                return a;
117        }
