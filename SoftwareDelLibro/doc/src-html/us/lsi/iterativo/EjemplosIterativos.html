003import java.util.Arrays;
004import java.util.List;

006import com.google.common.collect.Lists;

008public class EjemplosIterativos {
009
010        public static int binom(int n, int k) {
011                List<Integer> lsa = Arrays.asList(1);
012                int i = 1;
013                while (i <= n) {
014                        List<Integer> ls = Lists.newArrayList();
015                        for (int s = 0; s <= i; s++) {
016                                if (s == 0 || s == i) {
017                                        ls.add(1);
018                                } else if (s == 1 || s == i - 1) {
019                                        ls.add(i);
020                                } else {
021                                        ls.add(lsa.get(s - 1) + lsa.get(s));
022                                }
023                        }
024                        i = i + 1;
025                        lsa = Lists.newArrayList(ls);
026                }
027                return lsa.get(k);
028        }
029        
030        public static long fib(int n){
031                        int i = 0;
032                    int a = 1;
033                        int b = 0;
034                        while(i < n){
035                                i = i+1;
036                                int a0 = a;
037                                a = a0+b;
038                                b = a0;
039                        }
040                        return b;
041        }
042        
043        public static long pot(int b,int n){
044                int e = b;
045                int a = 1;
046                while( n > 0){
047                        if(n%2==1){
048                             a = a * e;
049                        }
050                        e = e * e;
051                        n = n/2;
052                }
053                return a;
054        }
055
056        public static long mcd(int a,int b) {
057                while(b > 0){
058                   int a0 = a;
059                   a = b;
060                   b = a0%b;   
061            }
062                return a;
063        }
064        public static void main(String[] args) {
065                System.out.println(binom(10,5));
066                System.out.println(fib(10));
067                System.out.println(pot(8,5));
068                System.out.println(mcd(10546,3280));
069        }

071}
