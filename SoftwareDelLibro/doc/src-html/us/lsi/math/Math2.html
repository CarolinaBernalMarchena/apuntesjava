049                MatrizFibonacci r;              
050                if(n==0){
051                        r = MatrizFibonacci.getOne();
052                }else if(n==1){
053                        r = MatrizFibonacci.getBase();
054                }else {
055                        r = powMatrizFibonacci(n/2);
056                        r = r.getSquare();

065         * @pre n &gt; = 0
066         * @param n Un Entero
067         * @return El número n de Fibonacci calculado reduciendo el problema a una potenciación de matrices
068         * y en una versión iterativa
069         */
070        public static BigInteger fibonacci5(Integer n){
071                BigInteger ar,br;
072                BigInteger au,bu;
073                BigInteger at,bt;
074                BigInteger dos = new BigInteger("2");
075                ar = BigInteger.ONE;
076                br = BigInteger.ZERO;
077                au = BigInteger.ZERO;
078                bu = BigInteger.ONE;
079                while(n>0){
080                        if(n%2==1){
081                                at = bu.multiply(ar).add(au.multiply(ar)).add(au.multiply(br));
082                                bt = au.multiply(ar).add(bu.multiply(br));
083                                au = at;
084                                bu = bt;
085                        }
086                        at = ar.multiply(ar).add(dos.multiply(ar).multiply(br));
087                        bt = ar.multiply(ar).add(br.multiply(br));
088                        ar = at;
089                        br = bt;
090                        n =n/2;
091                }
092                return au;
093        }

096         * @pre n &gt; = 0
097         * @param n Un Entero
098         * @return El número n de Fibonacci calculado reduciendo el problema a una potenciación de matrices
099         * y en una versión iterativa y usando Long en vez de BigInteger
100         */
101        public static Long fibonacci3(Integer n){
102                Long ar,br;
103                Long au,bu;
104                Long at,bt;
105                ar = 1L;
106                br = 0L;
107                au = 0L;
108                bu = 1L;
109                while(n>0){
110                        if(n%2==1){
111                                at = bu*ar+au*ar+au*br;
112                                bt = au*ar+bu*br;
113                                au = at;
114                                bu = bt;
115                        }
116                        at = ar*ar+2*ar*br;
117                        bt = ar*ar+br*br;
118                        ar = at;
119                        br = bt;
120                        n =n/2;
121                }
122                return au;
123        }

175                Double r;               
176                if(n==0){
177                        r = 1.;
178                }else if(n==1){
179                        r = base;

181                        r = powr(base,n/2);
182                        r = r*r;
183                        if(n%2!=0){
184                                r = r*base;
185                        }
186                }
187                return r;

193         * @param b Un segundo entero
194         * @return Calcula el máximo común divisor de los valores ablsolutos de los parámetros
195         */
196        public static int mcd(int a, int b){            
197               int u = Math.abs(a);
198               int v = Math.abs(b);
199               int r;
200               while (v != 0) {
201                    r = u % v;
202                    u = v;
203                    v = r;
204               }
205                    return u;
206        }

230                Integer valor;
231                        Preconditions.checkArgument(b>a,a+","+b);
232                if(b-a == 1){
233                        valor = a;
234                }else{

275         * @param increment El incremento &delta;
276         * @param t Temperatura
277         * @return Si &delta; &lt; 0 devuelve 1, si t = 0 devuelve 0, en otro caso devuelve e &#94; (- &delta;/t) 
278         */
279        public static double boltzmann(double increment, double t){
280                double r;
281                if(increment < 0.){
282                        r = 1.;
283                } else if(t==0.){
284                        r = 0.;
285                } else {
286                        r = Math.exp(-increment/t);
287                }
288                return r;

304         * @param probabilities Es una distribución de probabilidades para una variables aleatoria 
305         * con valores 0 hasta probabilidades.size() no incluido.
306         * @return Un entero entre 0 y probabilidades.size(), no incluido, con las probababilidades proporcionadas
307         */
308        public static Integer escogeEntre(List<Double> probabilities){
309                Preconditions.checkArgument(!probabilities.isEmpty());
310                Double ppa = 0.;
311                Integer r = 0;
312                double na = Math2.getDoubleAleatorio(0., 1.);
313                for(Double p:probabilities){
314                        ppa = ppa+p;
315                        if(ppa >= na){
316                                break;
317                        }
318                        r++;
319                }
320                return r;
321        }

488         * @pre n*nBits = ls.size()
489         * @param n Numero de enteros a decodificar
490         * @param nBits Numero de bits por entero.
491         * @param ls Una lista de bits
492         * @return Los numero enteros correspondientes a las n sublistas definidas 
493         */
494        public static  List<Integer> decodes(List<Integer> ls, Integer n, Integer nBits){
495                Preconditions.checkArgument(n*nBits == ls.size());
496                List<Integer> lsr = new ArrayList<>();
497                for (int i = 0; i < ls.size(); i = i + nBits) {
498                        Integer r = 0;
499                        for (Integer e : ls.subList(i,i+nBits)) {
500                                r = r * 2 + e;
501                        }
502                        lsr.add(r);
503                }
504                return lsr;
505        }

507        private static Integer[] pow2 = {1,2,4,8,16,32,64,128,256,512,1024,2048,4096,8192,16384,32768,65536,131072,262144,524288};
508        
509        /**
510         * @pre n*nBits = ls.size(), n = maxRanges.size()
511         * @pos return.size() = n, 0 &le; return[i] &lt; maxRanges(i)
512         * @param n Numero de enteros a decodificar
513         * @param nBits Numero de bits por entero.
514         * @param ls Una lista de bits
515         * @param maxRanges Una lista con los rangos máximos de los enteros resultantes
516         * @return Los numero enteros correspondientes a las n sublistas definidas en una escala con maximo maxRanges(i)-1
517         */
518        public static  List<Integer> decodesInScala(List<Integer> ls, Integer n, Integer nBits, List<Integer> maxRanges){
519                Preconditions.checkArgument(n*nBits == ls.size());
520                Preconditions.checkArgument(maxRanges.size()==n);
521                int maxEscala = nBits<20?pow2[nBits]:Math2.pow(2, nBits).intValue();
522                List<Integer> lsr = decodes(ls,n,nBits);
523                List<Integer> r = new ArrayList<>();
524                for(int i =0; i<n;i++){
525                        r.add(escala(lsr.get(i),maxEscala,maxRanges.get(i)));
526                }
527                return r;
528        }

534        public static Integer numeroDeBits(Integer e){
535                      int bits_necesarios = 0;
536                      while(e >=2) {
537                            bits_necesarios++;

539                      }

543        /**
544         * @param e Un entero positivo
545         * @return Una lista con la codificación en binario
546         */
547        public static List<Integer> code(Integer e){
548                      List<Integer> r = new ArrayList<>();
549                      while(e >=2) {
550                           if(e%2==0)
551                                   r.add(0,0);
552                           else
553                                   r.add(0,1);
554                           e = e/2;
555                      }
556                      r.add(0,e);                     
557                      return r;
558        }
