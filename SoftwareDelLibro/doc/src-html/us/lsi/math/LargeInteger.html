020        public static LargeInteger create(String digits,Integer base) {
021                boolean isPositive = true;
022                if(digits.charAt(0) == '-'){
023                        digits = digits.substring(1);
024                        isPositive = false;
025                }
026                List<Integer> dg = digits.chars().map(x->x-'0').boxed().collect(Collectors.toList());
027                LargeInteger r = eval(dg, base);
028                if (isPositive) return r;
029                else return r.negate();
030        }

080        private LargeInteger(List<Integer> digits, Boolean isPositive) {
081                super();
082                Preconditions.checkArgument(digits.size()>=1, "Debe haber un dígito al menos");
083                Preconditions.checkArgument(digits.stream().allMatch(x->x>=0), "Los digitos deben se positivos o cero");
084                if(digits.size() == 1 && digits.get(0).equals(0)) isPositive = true;
085                this.digits = Lists2.newList(digits);
086                this.isPositive = isPositive;
087        }

090        private static LargeInteger eval(List<Integer> n, Integer base){
091                LargeInteger a = ZERO;
092                LargeInteger b = ONE;
093                for(int i=n.size()-1; i >= 0; i--){
094                        a = a.add(b.multiply(create(n.get(i))));
095                        b = b.multiply(TEN);
096                }
097                return a;
098        }

191        public static int compare(LargeInteger a, LargeInteger b){
192                if(!a.isNegative() && b.isNegative()) return 1;
193                if(a.isNegative() && !b.isNegative()) return -1;
194                int r = 1;
195                if(a.isNegative() && b.isNegative()){
196                        r = -1; 
197                }
198                int mx = Math.max(a.size(),b.size());
199                a = completeWithLeftZeros(a,mx);
200                b = completeWithLeftZeros(b,mx);
201                int k = a.size();
202                Long ac = 0L;
203                for (int i = 0; i <k; i++) {
204                        ac = ac + a.getDigit(i) - b.getDigit(i);
205                        if (ac != 0) break;
206                }
207                return r*ac.intValue();
208        }

211                Long ab = a.longValue()+b.longValue();
212                Integer[] r = {(int)(ab/radix),(int)(ab%radix)};

216        private static Integer[] minusDigit(Integer a, Integer b){
217                Integer ab = a-b;
218                Integer r = 0;
219                if(ab<0) {
220                        ab = (int)(radix-b+a);
221                        r = -1;
222                }
223                Integer[] s = {r,ab};
224                return s;                               
225        }
226        private static Integer[] multiplyDigit(Integer a, Integer b){
227                Long ab = a.longValue()*b.longValue();
228                Integer[] s = {(int)(ab/radix),(int)(ab%radix)};
229                return s;                               
230        }

232        private static Integer[] divideDigit(Integer[] a, Integer b){
233                Preconditions.checkArgument(a.length ==2);
234                Long n = a[0].longValue()*radix+a[1];
235                Integer[] s = {(int)(n/b),(int)(n%b)};
236                return s;

239        public static LargeInteger add(LargeInteger a, LargeInteger b){
240                Boolean aPositive = true;
241                Boolean bPositive = true;
242                if(a.isNegative()){
243                        a = a.abs();
244                        aPositive = false;
245                }
246                if(b.isNegative()){
247                        b = b.abs();
248                        bPositive = false;
249                }
250                if(aPositive && !bPositive)
251                        return minus(a,b);
252                else if(!aPositive && !bPositive)
253                        return add(a,b).negate();
254                else if(!aPositive && bPositive)
255                        return minus(b,a);
256                
257                int mx = Math.max(a.size(),b.size());
258                a = completeWithLeftZeros(a,mx);
259                b = completeWithLeftZeros(b,mx);
260                int i =  a.size()-1;
261                int q = 0;
262                List<Integer> digits = Lists2.newList();
263                while(i >=0){
264                        Integer[] sd = addDigit(a.getDigit(i),q);
265                        q = sd[0];
266                        sd = addDigit(sd[1],b.getDigit(i));
267                        q = q+sd[0];
268                        digits.add(0,sd[1]);            
269                        i--;
270                }
271                digits.add(0,q);
272                return LargeInteger.create(digits).deleteNonSignificantZeros();
273        }

283        public static LargeInteger minus(LargeInteger a, LargeInteger b) {      
284                int mx = Math.max(a.size(), b.size());
285                a = completeWithLeftZeros(a, mx);
286                b = completeWithLeftZeros(b, mx);
287                Boolean positive = true;
288                int cmp = compare(a, b);
289                if(cmp == 0) return  ZERO;
290                if (cmp<0) {
291                        LargeInteger c = b; b = a; a = c;
292                        positive = false;
293                }
294                int i = a.size() - 1;
295                int q = 0;
296                List<Integer> digits = Lists2.newList();
297                while (i >= 0) {
298                        Integer[] sd = addDigit(a.getDigit(i),q);
299                        sd = minusDigit(sd[1],b.getDigit(i));
300                        q = sd[0];
301                        digits.add(0,sd[1]);            
302                        i--;
303                }
304                LargeInteger r = LargeInteger.create(digits).deleteNonSignificantZeros();
305                if(positive) return r;
306                else return r.negate();
307        }

317        public static LargeInteger multiply(LargeInteger m, LargeInteger n) {
318                Boolean aPositive = true;
319                Boolean bPositive = true;
320                Boolean rPositive = true;
321                if (m.isNegative()) {
322                        m = m.abs();
323                        aPositive = false;
324                }
325                if (n.isNegative()) {
326                        n = n.abs();
327                        bPositive = false;
328                }
329                if ((aPositive && !bPositive) || (!aPositive && bPositive))
330                        rPositive = false;
331                int mx = Math.max(m.size(), n.size());
332                LargeInteger r;
333                if (mx == 1) {
334                        Integer[] r0 = LargeInteger.multiplyDigit(m.getDigit(0),n.getDigit(0));
335                        r = LargeInteger.create(Arrays.asList(r0));
336                } else {
337                        if(mx%2 == 1) mx = mx+1;
338                        m = completeWithLeftZeros(m, mx);
339                        n = completeWithLeftZeros(n, mx);
340                        int k = mx / 2;
341                        LargeInteger m1 = LargeInteger.create(m.digits.subList(0, k));
342                        LargeInteger m0 = LargeInteger.create(m.digits.subList(k, mx));
343                        LargeInteger n1 = LargeInteger.create(n.digits.subList(0, k));
344                        LargeInteger n0 = LargeInteger.create(n.digits.subList(k, mx));
345                        LargeInteger z2 = m1.multiply(n1);
346                        LargeInteger z1 = m1.multiply(n0).add(m0.multiply(n1));
347                        LargeInteger z0 = m0.multiply(n0);              
348                        r = z2.multiplyRadixToM(2 * k).add(z1.multiplyRadixToM(k)).add(z0);
349                }
350                
351                if(rPositive) return r.deleteNonSignificantZeros();
352                else return r.negate().deleteNonSignificantZeros();
353        }

359        private static LargeInteger[] divide(LargeInteger a, Integer b){
360                LargeInteger[] zr1 = {ZERO,a};
361                LargeInteger[] zr2 = {a,ZERO};
362                if(a.equals(ZERO)) return zr1;
363                if(b.equals(ONE)) return zr2;   
364                Boolean aPositive = true;
365                Boolean bPositive = true;
366                if(a.isNegative()){
367                        a = a.abs();
368                        aPositive = false;
369                }
370                if(b<0){
371                        b = Math.abs(b);
372                        bPositive = false;
373                }
374                Boolean sigr = true;
375                if((aPositive && !bPositive) || (!aPositive && bPositive))
376                        sigr=false;
377                int k = a.size();
378                List<Integer> digits = Lists2.newList();        
379                Integer[] cp = {a.getDigit(0),0};
380                for(int i=0;i<k;i++){
381                        Integer[] s = {cp[1],a.getDigit(i)};
382                        cp = divideDigit(s,b);
383                        digits.add(cp[0]);              
384                }
385                LargeInteger c;         
386                if(sigr) c = LargeInteger.create(digits).deleteNonSignificantZeros();
387                else c = LargeInteger.create(digits).deleteNonSignificantZeros().negate();
388                LargeInteger[] r = {c,create(cp[1])};
389                return r;
390        }

392        public static LargeInteger[] divide(LargeInteger num, LargeInteger den){
393                Boolean numPositive = true;
394                Boolean denPositive = true;
395                if(num.isNegative()){
396                        num = num.abs();
397                        numPositive = false;
398                }
399                if(den.isNegative()){
400                        den = den.abs();
401                        denPositive = false;
402                }
403                Boolean sigr = true;
404                if((numPositive && !denPositive) || (!numPositive && denPositive))
405                        sigr=false;
406                
407                if(compare(num,den)<0){
408                        LargeInteger[] ss ={ZERO,den};
409                        return ss;
410                } 
411                
412                int kd = den.size();
413                Integer denD = den.getDigit(0); 
414                LargeInteger numD = num.divideRadixToM(kd-1);
415                LargeInteger q = divide(numD,denD)[0];
416                LargeInteger r = add(den,ONE);
417                LargeInteger s = TWO;
418                LargeInteger t = TWO;
419                while(compare(r.abs(),den)>=0 && compare(t.abs(),ONE)> 0){
420                        r = num.minus(q.multiply(den));
421                        s = divide(r.divideRadixToM(kd-1),denD)[0];
422                        t = divide(s, 2)[0];                    
423                        q = add(q, t);
424                }
425                r = minus(num,multiply(q,den));
426                while(r.isNegative()){
427                        q = q.minus(ONE);
428                        r = r.add(den);
429                }
430                q = LargeInteger.create(q.digits, sigr);
431                LargeInteger[] u = {q,r};
432                return u;
433        }

443        @Override
444        public String toString() {
445                String r = isPositive?"":"-";
446                LargeInteger n = this;
447                StringBuffer rst = new StringBuffer();          
448                while(n.compareTo(TEN) >= 0){
449                        LargeInteger[] cr = n.divide(10);
450                        n = cr[0];
451                        rst.insert(0,cr[1].getDigit(0));
452                }
453                rst.insert(0,n.getDigit(0));
454                return r+rst.toString();
455        }
