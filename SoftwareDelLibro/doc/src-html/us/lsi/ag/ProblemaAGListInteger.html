001package us.lsi.ag;
002
003import java.util.List;
004
005import org.apache.commons.math3.genetics.CrossoverPolicy;
006import org.apache.commons.math3.genetics.MutationPolicy;
007
008import us.lsi.ag.agchromosomes.ListIntegerChromosome;
009
010/**
011 * @author Miguel Toro
012 *
013 * @param <S>
014 * 
015 * <p> Es un cromosoma general para trabajar con problemas cuyas soluciones pueden ser modeladas por listas de enteros.</p>
016 */
017public interface ProblemaAGListInteger<S> extends ProblemaAG {
018
019        /**
020         * @return La dimensión de la lista de enteros
021         */
022        int getDimension();
023
024        /**
025         * @param listIntegerChromosome Un cromosoma
026         * @return la función de fitness
027         */
028        double fitnessFunction(ListIntegerChromosome listIntegerChromosome);
029
030        /**
031         * @return Una lista aleatoria de enteros de tamaño getDimensión() que cumpla los requisitos del problema 
032         */
033        List<Integer> getRandomList();
034        
035        /**
036         * @return El operador de Mutación usado
037         */
038        MutationPolicy getMutationPolicy();
039
040        /**
041         * @return El operador de mezcla (crossover)
042         */
043        CrossoverPolicy getCrossoverPolicy();
044        
045        /**
046         * @param c Un cromosoma
047         * @return La solución asociada
048         */
049        S getSolucion(ListIntegerChromosome c);
050}
