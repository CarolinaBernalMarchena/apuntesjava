001package us.lsi.ag;
002
003import java.util.List;
004import java.util.stream.Collectors;
005import java.util.stream.IntStream;
006
007import com.google.common.base.Preconditions;
008
009import us.lsi.common.Lists2;
010/**
011 * @author Miguel Toro
012 *
013 * @param <S> Tipo de la solución
014 * 
015 * 
016 * <p> Se parte de un conjunto de <code> n </code> objetos distintos y unas multiplicidades máximas para cada uno de ellos
017 * A partir de esa información se construye la secuencia normal que asumimos de tamaño <code> r </code>. </p>
018 * 
019 * <p> La secuencia normal asociada al problema. Siendo <code> n </code> el número de objetos la secuencia normal está formado por la concatenación de  
020 * <code> n </code> sublistas <code> L(i) </code>. Cada <code> L(i) </code> está formada por <code> getMax(i) </code> copias del entero <code> i </code>.
021 * con <code> i </code> en el rango <code> 0..n-1 </code>. </p>
022 *  
023 * <p> Los problemas adecuados para ser modelados con este tipo son aquellos cuya solución 
024 * es un multiconjunto o una lista, posiblemente con repetición, de los objetos dados. </p>
025 */
026public interface IndexProblemAG<S> extends ProblemAG {  
027        
028        
029        /**
030         * @return Numero de objetos distintos
031         */
032        Integer getObjectsNumber();
033        
034        /**
035         * @pre <code> 0 &le; index &lt; getObjetos().size() </code>
036         * @param index Indice en la lista de objetos disponibles
037         * @return La multiplicidad máxima del objeto de índice <code> index </code>. 
038         * La multiplicidad máxima del objeto <code> i </code> estará en el rango <code> 0..getMax(i) </code>
039         */
040        
041        default Integer getMaxMultiplicity(int index){
042                Preconditions.checkElementIndex(index, this.getObjectsNumber());
043                return 1;
044        }
045        
046        
047        
048        /**
049         * @param cr Un cromosoma
050         * @return La función de fitnes del cromosoma
051         */
052        Double fitnessFunction(IndexChromosome cr);
053        
054        /**
055         * @param cr Un cromosoma
056         * @return La solución definida por el cromosoma
057         */
058        S getSolucion(IndexChromosome cr);
059        
060        
061        
062    /**
063     * @return La secuencia normal asociada al problema. Siendo <code> n </code> el número de objetos la secuencia normal está formado por la concatenación de  
064     * <code> n </code> sublistas <code> L(i) </code>. Cada <code> L(i) </code> está formada por <code> getMax(i) </code> copias del entero <code> i </code>.
065     * con <code> i </code> en el rango <code> 0..n-1 </code>.
066     */
067    default List<Integer> getNormalSequence() {
068                return IntStream.range(0,getObjectsNumber())
069                                .boxed()
070                                .map(x->Lists2.nCopias(getMaxMultiplicity(x),x).stream())
071                                .flatMap(x->x)
072                                .collect(Collectors.toList());
073        }
074}
