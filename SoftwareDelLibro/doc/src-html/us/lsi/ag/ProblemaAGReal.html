001package us.lsi.ag;
002
003import java.util.List;
004
005import us.lsi.ag.agchromosomes.IRealChromosome;
006import us.lsi.common.Tuple2;
007
008/**
009 * @author Miguel Toro
010 *
011 * @param <S> Tipo de la solución
012 * 
013 * <p> Tipo adecuado para encontrar el máximo de una función de varias varaibles reales en un dominio dado </p>
014 */
015public interface ProblemaAGReal<S> extends ProblemaAG {
016        
017        /**
018         * @return Número de variables
019         */
020        Integer getNumeroDeVariables(); 
021        
022        /**
023         * @return Límites para cada variable
024         */
025        List<Tuple2<Double,Double>> getLimites();
026        
027        /**
028         * @param i El indice de una variable
029         * @return El límite inferior de la variable i
030         */
031        default Double geLimiteInf(int i){
032                return this.getLimites().get(i).v1;
033        }
034                
035        /**
036         * @param i El indice de una variable
037         * @return El límite superior de la variable i
038         */
039        default Double geLimiteSup(int i){
040                return this.getLimites().get(i).v2;
041        }
042        
043        /**
044         * @param chromosome Un cromosoma
045         * @return La función de fitness del cromosoma
046         */
047        Double fitnessFunction(IRealChromosome chromosome);
048        
049        /**
050         * @param chromosome Es un cromosoma
051         * @return La solución definida por el cromosoma
052         */
053        S getSolucion(IRealChromosome chromosome);
054        
055}
