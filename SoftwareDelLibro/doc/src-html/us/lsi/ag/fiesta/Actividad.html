001package us.lsi.ag.fiesta;
002
003public class Actividad {
004        private Double valoracion;
005        private Double coste;
006        private Integer id;
007        
008        public Actividad(Double valoracion, Double coste) {
009                super();
010                this.valoracion = valoracion;
011                this.coste = coste;
012        }
013        public Double getValoracion() {
014                return valoracion;
015        }
016        public Double getCoste() {
017                return coste;
018        }
019        public Integer getId() {
020                return id;
021        }
022        public void setId(Integer id) {
023                this.id = id;
024        }
025        @Override
026        public int hashCode() {
027                final int prime = 31;
028                int result = 1;
029                result = prime * result + ((coste == null) ? 0 : coste.hashCode());
030                result = prime * result + ((id == null) ? 0 : id.hashCode());
031                result = prime * result
032                                + ((valoracion == null) ? 0 : valoracion.hashCode());
033                return result;
034        }
035        @Override
036        public boolean equals(Object obj) {
037                if (this == obj)
038                        return true;
039                if (obj == null)
040                        return false;
041                if (!(obj instanceof Actividad))
042                        return false;
043                Actividad other = (Actividad) obj;
044                if (coste == null) {
045                        if (other.coste != null)
046                                return false;
047                } else if (!coste.equals(other.coste))
048                        return false;
049                if (id == null) {
050                        if (other.id != null)
051                                return false;
052                } else if (!id.equals(other.id))
053                        return false;
054                if (valoracion == null) {
055                        if (other.valoracion != null)
056                                return false;
057                } else if (!valoracion.equals(other.valoracion))
058                        return false;
059                return true;
060        }
061        @Override
062        public String toString() {
063                return "Actividad [valoracion=" + valoracion + ", coste=" + coste
064                                + ", id=" + id + "]";
065        }
066        
067        
068        
069}
