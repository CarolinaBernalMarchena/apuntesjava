001package us.lsi.ag;
002
003import java.util.List;
004
005import us.lsi.ag.agchromosomes.ExpressionChromosome;
006import us.lsi.tiposrecursivos.AccumulatorExp;
007import us.lsi.tiposrecursivos.Operator;
008
009/**
010 * @author Miguel Toro
011 *
012 * @param <S> Tipo de la solución
013 * @param <T> Tipo del resultado de la expresión buscada
014 * 
015 * <p> Tipo adecuado para modelar problemas dónde queremos encontrar una expresión, 
016 * formada por un conjunto de operadores dados, un conjunto de variables y un conjunto de constantes, tal que su valor se ajuste 
017 * a algunos requisitos dados </p>
018 */
019public interface ExpressionProblemAG<S,T> extends ProblemAG {
020
021        
022        /**
023         * @return La longitud de la cabeza de un gen
024         */
025        Integer getHeadLength();
026        
027        /**
028         * @return Número de Genes
029         */
030        Integer getNumGenes();
031        
032        /**
033         * @return Número de variables
034         */
035        Integer getNumVariables();
036        
037        /**
038         * @return Numero de constantes
039         */
040        Integer getNumConstants();
041        
042        /**
043         * @return El rango máximo del valor de cada constante. Cada constante tendrá un valor en el rango 0..getMaxValueConstant()-1
044         */
045        Integer getMaxValueConstant();
046        
047        /**
048         * @return Número de operadores
049         */
050        List<Operator> getOperators();  
051        
052        /**
053         * @return Expresión n-aria, de aridad variable, que combinará las expresiones asociadas a cada uno de los genes del cromosoma
054         */
055        AccumulatorExp<T,T> getAccumulator();
056        
057        /**
058         * @param e Un número entero
059         * @return Un valor de tipo T resultado de la conversión del entero e.
060         */
061        T convert(Integer e);
062        
063        
064        /**
065         * @param chromosome Un cromosoma 
066         * @return La función de fitness del cromosoma
067         */
068        Double fitnessFunction(ExpressionChromosome<T> chromosome);
069        
070        /**
071         * @param chromosome Es un cromosoma
072         * @return La solución definida por el cromosoma
073         */
074        S getSolucion(ExpressionChromosome<T> chromosome);
075}
