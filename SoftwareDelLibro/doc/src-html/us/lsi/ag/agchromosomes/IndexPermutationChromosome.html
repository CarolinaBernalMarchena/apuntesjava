001package us.lsi.ag.agchromosomes;
002
003import java.util.List;
004
005import org.apache.commons.math3.genetics.AbstractListChromosome;
006import org.apache.commons.math3.genetics.InvalidRepresentationException;
007import org.apache.commons.math3.genetics.RandomKey;
008
009import us.lsi.ag.IndexChromosome;
010import us.lsi.ag.ProblemAG;
011import us.lsi.ag.IndexProblemAG;
012
013
014import com.google.common.base.Preconditions;
015
016/**
017 * @author Miguel Toro
018 * 
019 * <p> Una implementación del tipo IndexChromosome. Toma como información la definición de un problema que implementa el interfaz IndexProblemAG.
020 * A partir de esa información construye una secuencia normal. 
021 * Asumimos que el número de objetos es <code>n </code>y el tamaño de la secuencia normal <code>r</code>. 
022 * Lista decodificada es una permutación de la secuencia normal.</p>
023 *  
024 * <p> La lista decodificada está formada por una lista de  tamaño <code>r</code>, cuyos valores son 
025 * índices en el rango <code> [0,n-1]</code>, y cada índice <code>i</code> se  repite un número de veces igual al 
026 * dado por la multiplicidad máxima del objeto <code> i </code>
027 * definida en el problema. </p>
028 * 
029 * <p> La implementación usa un cromosoma de clave aleatoria de tamaño <code> r </code>.
030 * Es un cromosoma adecuado para codificar problemas de permutaciones </p>
031 *
032 */
033public class IndexPermutationChromosome extends RandomKey<Integer> implements IndexChromosome {
034
035        public static List<Integer> normalSequence = null;
036        public static IndexProblemAG<?> problema;
037        
038        /**
039         * Dimensión del cromosoma
040         */
041        
042        public static int DIMENSION;
043        
044        public static void iniValues(ProblemAG problema){
045                IndexPermutationChromosome.problema = (IndexProblemAG<?>) problema; 
046                IndexPermutationChromosome.normalSequence = IndexPermutationChromosome.problema.getNormalSequence();
047                IndexPermutationChromosome.DIMENSION = IndexPermutationChromosome.normalSequence.size();
048        }
049
050        
051        public IndexPermutationChromosome(List<Double> representation)
052                        throws InvalidRepresentationException {
053                super(representation);
054                this.ft = this.calculateFt();
055        }
056
057        public IndexPermutationChromosome(Double[] representation)
058                        throws InvalidRepresentationException {
059                super(representation);
060                this.ft = this.calculateFt();
061        }
062
063        @Override
064        public AbstractListChromosome<Double> newFixedLengthChromosome(List<Double> ls) {
065                return new IndexPermutationChromosome(ls);
066                
067        }
068
069        @Override
070        public List<Integer> decode() {
071                Preconditions.checkArgument(IndexPermutationChromosome.normalSequence!=null);
072                return this.decode(IndexPermutationChromosome.normalSequence);
073        }
074        
075        
076        public static IndexPermutationChromosome getInitialChromosome() {
077                List<Double> ls = RandomKey.randomPermutation(IndexPermutationChromosome.DIMENSION);
078                return new IndexPermutationChromosome(ls);
079        }
080
081        
082        @Override
083        public double fitness() {
084                return ft;
085        }
086        
087        private double ft;
088        
089        private double calculateFt(){
090                return IndexPermutationChromosome.problema.fitnessFunction(this);
091        }
092
093        @Override
094        public IndexProblemAG<?> getProblem() {
095                return IndexPermutationChromosome.problema;
096        }
097
098        public Integer getObjectsNumber() {
099                return IndexPermutationChromosome.problema.getObjectsNumber();
100        }
101
102        public Integer getMax(int i) {
103                return IndexPermutationChromosome.problema.getMaxMultiplicity(i);
104        }
105        
106}
