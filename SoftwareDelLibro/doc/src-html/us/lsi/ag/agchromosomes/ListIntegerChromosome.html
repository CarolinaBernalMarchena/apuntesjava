001package us.lsi.ag.agchromosomes;
002
003import java.util.List;
004
005import org.apache.commons.math3.genetics.AbstractListChromosome;
006import org.apache.commons.math3.genetics.InvalidRepresentationException;
007
008import us.lsi.ag.Chromosome;
009import us.lsi.ag.ProblemAG;
010import us.lsi.ag.ProblemaAGListInteger;
011
012/**
013 * @author Miguel Toro
014 * 
015 * <p> Una implementación del tipo IChromosome&lt;Integer&gt;. Toma como información la definición de un problema que implementa el interfaz ProblemaAGListInteger. </p>
016 *  
017 * <p> La lista decodificada está formada por una lista de  entero de tamaño r </p>
018 * 
019 * <p> La implementación usa una lista de enteros de tamaño r.
020 * Es un cromosoma general para trabajar con problemas cuyas soluciones pueden ser modeladas por listas de enteros. </p>
021 *
022 */
023
024public class ListIntegerChromosome extends AbstractListChromosome<Integer> implements Chromosome<List<Integer>> {
025
026        public static ProblemaAGListInteger<?> problema;
027        
028        /**
029         * Dimensión del cromosoma
030         */
031        
032        public static int DIMENSION;
033        
034        public static void iniValues(ProblemAG problema){
035                ListIntegerChromosome.problema = (ProblemaAGListInteger<?>) problema; 
036                ListIntegerChromosome.DIMENSION = ListIntegerChromosome.problema.getDimension();
037        }
038        
039        public ListIntegerChromosome(List<Integer> representation) throws InvalidRepresentationException {
040                super(representation);
041                this.ft = this.calculateFt();
042        }
043
044        public ListIntegerChromosome(Integer[] representation) throws InvalidRepresentationException {
045                super(representation);
046                this.ft = this.calculateFt();
047        }       
048        
049        @Override
050        public List<Integer> decode() {
051                return super.getRepresentation();
052        }
053
054        public static ListIntegerChromosome getInitialChromosome() {
055                List<Integer> ls = ListIntegerChromosome.problema.getRandomList();
056                return new ListIntegerChromosome(ls);
057        }
058        
059        private double ft;
060        
061        private double calculateFt(){
062                return ListIntegerChromosome.problema.fitnessFunction(this);
063        }
064
065        @Override
066        public double fitness() {
067                return ft;
068        }
069
070        @Override
071        protected void checkValidity(List<Integer> arg0) throws InvalidRepresentationException {
072
073        }
074
075        @Override
076        public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> ls) {
077                return new ListIntegerChromosome(ls);
078        }
079
080        
081
082}
