032        /**
033         * Los diferentes tipos de cromosmomas implementados
034         *
035         */
036        public enum ChromosomeType {Binary,ListInteger,IndexSubList,IndexRange,IndexPermutation,IndexPermutationSubList,Real,Expression}
037        
038        public static ChromosomeType tipo;
039
040        /**
041         * @param tipo El tipo de cromosoma
042         * @return Un cromosoma aleatorio del tipo indicado
043         */
044        public static IChromosome<?> randomChromosome(ChromosomeType tipo){
045                IChromosome<?> chromosome = null;
046                switch(tipo){
047                case Binary: chromosome = BinaryChromosome.getInitialChromosome(); break;
048                case IndexSubList: chromosome = IndexSubListChromosome.getInitialChromosome(); break;
049                case ListInteger: chromosome = ListIntegerChromosome.getInitialChromosome(); break;
050                case IndexRange: chromosome = IndexRangeChromosome.getInitialChromosome(); break;
051                case IndexPermutation: chromosome = IndexPermutationChromosome.getInitialChromosome(); break;
052                case IndexPermutationSubList: chromosome = IndexPermutationSubListChromosome.getInitialChromosome(); break;
053                case Real: chromosome = RealChromosome.getInitialChromosome(); break;
054                case Expression: chromosome = ExpressionChromosome.getInitialChromosome(); break;
055                }
056                return chromosome;
057        }
058        
059        /**
060         * <p>Distintos tipo de operadores de cruce </p>
061         *
062         * <ul>
063         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/OnePointCrossover.html" target="_blank"> OnePointCrossover </a>
064         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/NPointCrossover.html" target="_blank"> NPointCrossover </a>
065         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/CycleCrossover.html" target="_blank"> CycleCrossover </a>
066         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/OrderedCrossover.html" target="_blank"> OrderedCrossover </a>
067         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/UniformCrossover.html" target="_blank"> UniformCrossover </a>
068         * </ul> 
069         * 
070         */
071        
072        public enum CrossoverType{Cycle,NPoint,OnePoint,Ordered,Uniform};
073        
074        /**
075         * Tipo del operador de cruce
076         */
077        public static CrossoverType crossoverType = CrossoverType.OnePoint;
078        
079        /**
080         * Número de puntos usados en la partición si se usa un operador de cruce de tipo NPointCrossover
081         */
082        public static int NPOINTCROSSOVER = 3;
083        /**
084         * La ratio si se usa el operador de cruce de tipo UniformCrossover
085         */
086        public static double RATIO_UNIFORMCROSSOVER = 0.7;
087        
088        /**
089         * @param tipo El tipo del cromosoma
090         * @param problema Las propiedades del probblema a resolver
091         * @return Un operador de cruce adecuado para un cromosma del tipo indicado
092         */
093        public static CrossoverPolicy getCrossoverPolicy(ChromosomeType tipo, ProblemaAG problema){
094                CrossoverPolicy crossOverPolicyBin = null;      
095                switch(crossoverType){
096                case Cycle: crossOverPolicyBin = new CycleCrossover<Integer>();break;
097                case NPoint: crossOverPolicyBin = new NPointCrossover<Integer>(NPOINTCROSSOVER);break;
098                case OnePoint: crossOverPolicyBin = new OnePointCrossover<Integer>();break;
099                case Ordered: crossOverPolicyBin = new OrderedCrossover<Integer>(); break;
100                case Uniform: crossOverPolicyBin = new UniformCrossover<Integer>(RATIO_UNIFORMCROSSOVER); break;
101                }
102                CrossoverPolicy crossOverPolicyKey = null;      
103                switch(crossoverType){
104                case Cycle: crossOverPolicyKey = new CycleCrossover<Double>();break;
105                case NPoint: crossOverPolicyKey = new NPointCrossover<Double>(NPOINTCROSSOVER);break;
106                case OnePoint: crossOverPolicyKey = new OnePointCrossover<Double>();break;
107                case Ordered: crossOverPolicyKey = new OrderedCrossover<Double>(); break;
108                case Uniform: crossOverPolicyKey = new UniformCrossover<Double>(RATIO_UNIFORMCROSSOVER); break;
109                }
110                CrossoverPolicy crossOverPolicy = null; 
111                switch(tipo){
112                case Binary: crossOverPolicy = crossOverPolicyBin; break;
113                case IndexSubList: crossOverPolicy = crossOverPolicyBin; break;
114                case ListInteger: crossOverPolicy = ((ProblemaAGListInteger<?>)problema).getCrossoverPolicy(); break;
115                case IndexRange: crossOverPolicy = crossOverPolicyBin; break;
116                case IndexPermutation: crossOverPolicy = crossOverPolicyKey; break;
117                case IndexPermutationSubList: crossOverPolicy = new SubListCrossoverPolicy(crossOverPolicyBin,crossOverPolicyKey); break;
118                case Real: crossOverPolicy = crossOverPolicyBin; break;
119                case Expression: crossOverPolicy = crossOverPolicyBin; break;
120                }
121                Preconditions.checkState(crossOverPolicy!=null);
122                return crossOverPolicy;
123        }
124        
125        /**
126         * @param tipo El tipo del cromosoma
127         * @param problema El problema a resolver
128         * @return Un operador de mutación adecuado para un cromosoma del tipo indicado
129         */
130        public static MutationPolicy getMutationPolicy(ChromosomeType tipo, ProblemaAG problema){
131                MutationPolicy mutationPolicy = null;
132                switch(tipo){
133                case Binary:  mutationPolicy = new BinaryMutation()     ; break;
134                case IndexSubList: mutationPolicy = new BinaryMutation(); break;
135                case ListInteger: mutationPolicy = ((ProblemaAGListInteger<?>)problema).getMutationPolicy(); break;
136                case IndexRange: mutationPolicy = new BinaryMutation(); ; break;
137                case IndexPermutation: mutationPolicy = new RandomKeyMutation(); break;
138                case IndexPermutationSubList: mutationPolicy = new SubListMutationPolicy(); break;
139                case Real: mutationPolicy = new BinaryMutation();       ; break;
140                case Expression: mutationPolicy = new BinaryMutation(); ; break;
141                }
142                Preconditions.checkState(mutationPolicy!=null);
143                return mutationPolicy;
144        }
145        
146        /**
147         * <p> Para aplicar los operadores de mutación se escogen dos cromosomas. 
148         * La técnica implementada para escoger cada uno de los dos cromosomas se denomina selección por torneo. 
149         * Se trata de organizar dos torneos. 
150         * En cada uno se elige el mejor cromosoma de entrre <code> TOURNAMENT_ARITY </code> cromosomas de la población seleccionados al azar. 
151         * Si el tamaño de <code> TOURNAMENT_ARITY </code> es más grande, los cromosomas
152         *  débiles tienen menor probabilidad de ser seleccionados.</p>
153         * 
154         * <p> Número de participantes en el torneo para elegir los cromosomas que participarán en el cruce </p>
155         * <p> Un valor típico es 2 </p>
156         */
157        
158        public static int TOURNAMENT_ARITY = 2;
159        
160        /**
161         * @return Un operador que implementa la política de selección escogida
162         */
163        public static SelectionPolicy getSelectionPolicy(){     
164                SelectionPolicy selectionPolicy = new TournamentSelection(TOURNAMENT_ARITY);
165                Preconditions.checkState(selectionPolicy!=null);
166                return selectionPolicy;
167        }

169        /**
170         * @param tipo Tipo de cromosoma
171         * @param problema El problema a resolver 
172         * @pos El método inicializa los parámetros relevantes de la clase que implementa el tipo indicado de cromosoma
173         */
174        public static void iniValues(ChromosomeType tipo, ProblemaAG problema){
175                switch(tipo){
176                case Binary: BinaryChromosome.iniValues(problema);break;
177                case IndexSubList: IndexSubListChromosome.iniValues(problema);break;
178                case ListInteger: ListIntegerChromosome.iniValues(problema);break;
179                case IndexRange: IndexRangeChromosome.iniValues(problema); break;
180                case IndexPermutation: IndexPermutationChromosome.iniValues(problema);break;
181                case IndexPermutationSubList: IndexPermutationSubListChromosome.iniValues(problema);break;
182                case Real: RealChromosome.iniValues(problema);break;
183                case Expression: ExpressionChromosome.iniValues(problema);break;
184                }
185        }
