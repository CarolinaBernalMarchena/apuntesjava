027/**
028 * 
029 * 
030 * 
031 * 
032 * @author Miguel Toro
033 * 
034 * 
035* <p> Una implementación del tipo IndexChromosome. Toma como información la definición de un problema que implementa el interfaz IndexProblemAG.
036 * A partir de esa información construye una secuencia normal. 
037 * Asumimos que el número de objetos es <code>n </code>y el tamaño de la secuencia normal <code>r</code>. 
038 * Lista decodificada es una permutación de un subconjunto de la secuencia normal.</p>
039 *  
040 * <p> La lista decodificada está formada, por tanto, 
041 *  por una lista de  tamaño menor o igual a <code>r</code>, cuyos valores son 
042 * índices en el rango <code> [0,n-1]</code>, y cada índice <code>i</code> se  repite un número de veces igual o menor 
043 * al dado por la multiplicidad máxima del objeto <code> i </code>
044 * definida en el problema. </p>
045 * 
046 * <p> La implementación usa un cromosoma de clave aleatoria de tamaño <code> r </code> y oro binario del 
047 * mismo tamaño.
048 * Es un cromosoma adecuado para codificar problemas de subcojuntos de permutaciones.</p>
049 */
050public class IndexPermutationSubListChromosome extends org.apache.commons.math3.genetics.Chromosome 
051
052                                                        implements IndexChromosome {
053
054        public static List<Integer> normalSequence;
055        
056        public static IndexProblemAG<?> problema;
057        
058        /**
059         * Dimensión del cromosoma
060         */
061        
062        protected static int DIMENSION;
063        
064        
065        public static void iniValues(ProblemAG problema){
066                IndexPermutationSubListChromosome.problema = (IndexProblemAG<?>) problema; 
067                IndexPermutationSubListChromosome.normalSequence = IndexPermutationSubListChromosome.problema.getNormalSequence();
068                IndexPermutationSubListChromosome.DIMENSION = IndexPermutationSubListChromosome.normalSequence.size();
069        }
070        
071        private BinaryChromosome2 binary;
072        private RandomKey2 randomKey;   
073        
074        
075        
076        /**
077         * @param binary Un cromosoma binario
078         * @param randomKey Un cromosoma randomKey
079         */
080        public IndexPermutationSubListChromosome(Chromosome binary, Chromosome randomKey) {
081                super();
082                if(binary instanceof BinaryChromosome2){
083                        this.binary = (BinaryChromosome2)binary;
084                }
085                if(randomKey instanceof RandomKey2){
086                        this.randomKey = (RandomKey2) randomKey;
087                }
088                Preconditions.checkArgument(this.binary!=null);
089                Preconditions.checkArgument(this.randomKey!=null);
090                Preconditions.checkArgument(this.binary.getLength()==this.randomKey.getLength());       
091                this.ft = this.calculateFt();
092        }
093        
094        /**
095         * Un constructor adecuado para crear un objeto por defecto de este tipo
096         */
097        public IndexPermutationSubListChromosome() {
098                super();
099                List<Integer> ls1 = BinaryChromosome2.randomBinaryRepresentation(100);
100                List<Double>  ls2 = RandomKey2.randomPermutation(100);
101                BinaryChromosome2 c1 = new BinaryChromosome2(ls1);
102                RandomKey2 c2 = new RandomKey2(ls2);
103                this.binary =  c1;
104                this.randomKey = c2;    
105                this.ft = 0.;
106        }
107        
108        /* (non-Javadoc)
109         * @see org.apache.commons.math3.genetics.Fitness#fitness()
110         */
111        
112        
113        @Override
114        public double fitness() {
115                return ft;
116        }
117        
118        private Double ft;
119        
120        private double calculateFt(){
121                return IndexPermutationSubListChromosome.problema.fitnessFunction(this);
122        }
123        
124        @Override
125        public IndexProblemAG<?> getProblem() {
126                return IndexPermutationSubListChromosome.problema;
127        }
128
129        public Integer getObjectsNumber() {
130                return IndexPermutationSubListChromosome.problema.getObjectsNumber();
131        }
132
133        public Integer getMax(int i) {
134                return IndexPermutationSubListChromosome.problema.getMaxMultiplicity(i);
135        }
136        
137        
138        public int compareTo(Chromosome another) {
139                if (!(another instanceof IndexPermutationSubListChromosome))
140                        throw new IllegalArgumentException();;
141                IndexPermutationSubListChromosome other = (IndexPermutationSubListChromosome) another;
142                Double f1 = this.fitness();
143                Double f2 = other.fitness();
144                return f1.compareTo(f2);
145        }
146        
147        /**
148         * @return Una lista de enteros obtenida permutando la secuencia normal (0, 1, 2, ..., r-1) filtrada para incluir 
149         * sólo los seleccionados por el cromosoma binario 
150         */
151        public List<Integer> decode() { 
152                List<Integer> rk = randomKey.decode(normalSequence);
153                List<Integer> r = Lists.newArrayList();
154                List<Integer> bn = binary.getRepresentation();
155                Preconditions.checkArgument(rk.size()==bn.size());
156                for(int i=0; i< rk.size();i++){
157                        if(bn.get(i)==1){
158                                r.add(rk.get(i));
159                        }
160                }
161                return r;
162        }
163
164        /**
165         * @return La dimensión del cromosoma
166         */
167        public int getLength() {
168                return randomKey.getLength();
169        }       
170        
171
172        @Override
173        public boolean equals(Object obj) {
174                if (this == obj)
175                        return true;
176                if (obj == null)
177                        return false;
178                if (!(obj instanceof IndexPermutationSubListChromosome))
179                        return false;
180                IndexPermutationSubListChromosome other = (IndexPermutationSubListChromosome) obj;
181                if (binary == null) {
182                        if (other.binary != null)
183                                return false;
184                } else if (!binary.equals(other.binary))
185                        return false;
186                if (randomKey == null) {
187                        if (other.randomKey != null)
188                                return false;
189                } else if (!randomKey.equals(other.randomKey))
190                        return false;
191                return true;
192        }       
193
194        @Override
195        public int hashCode() {
196                final int prime = 31;
197                int result = 1;
198                result = prime * result + ((binary == null) ? 0 : binary.hashCode());
199                result = prime * result
200                                + ((randomKey == null) ? 0 : randomKey.hashCode());
201                return result;
202        }
203
204        public String toString() {
205                return this.fitness()+","+this.decode();
206        }
207
208        
209        
210        public BinaryChromosome getBinary() {
211                return binary;
212        }
213
214        public RandomKey<Integer> getRandomKey() {
215                return randomKey;
216        }
217        
218        /**
219         * @param dimension La dimensión del cromosoma
220         * @pre Debe estar inicializada la propiedad factory
221         * @return Un cromosoma mixto aleatorio
222         * 
223         */
224        private static IndexPermutationSubListChromosome random(Integer dimension){
225                List<Integer> ls1 = BinaryChromosome2.randomBinaryRepresentation(dimension);
226                List<Double>  ls2 = RandomKey2.randomPermutation(dimension);
227                BinaryChromosome2 c1 = new BinaryChromosome2(ls1);
228                RandomKey2 c2 = new RandomKey2(ls2);
229                return new IndexPermutationSubListChromosome(c1, c2);
230        }
231        
232        public static IndexPermutationSubListChromosome getInitialChromosome() {
233                return IndexPermutationSubListChromosome.random(IndexPermutationSubListChromosome.DIMENSION);
234        }

236        private static class BinaryChromosome2 extends BinaryChromosome {               
237
238                public BinaryChromosome2(Integer[] representation)
239                                throws InvalidRepresentationException {
240                        super(representation);
241                }
242
243                public BinaryChromosome2(List<Integer> representation)
244                                throws InvalidRepresentationException {
245                        super(representation);
246                }
247
248                @Override
249                public double fitness() {
250                        return 0;
251                }
252
253                @Override
254                public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> ls) {
255                        return new BinaryChromosome2(ls);
256                }
257                 
258                @Override
259                public List<Integer> getRepresentation(){
260                        return super.getRepresentation();
261                }       
262        }

264        private static class RandomKey2 extends  RandomKey<Integer> {
265                
266                
267                public RandomKey2(Double[] representation) throws InvalidRepresentationException {
268                        super(representation);
269                }
270
271                public RandomKey2(List<Double> representation) throws InvalidRepresentationException {
272                        super(representation);
273                }
274
275                @Override
276                public double fitness() {
277                        return 0;
278                }
279
280                @Override
281                public AbstractListChromosome<Double> newFixedLengthChromosome(List<Double> ls) {
282                        return new RandomKey2(ls);
283                }
284        }
