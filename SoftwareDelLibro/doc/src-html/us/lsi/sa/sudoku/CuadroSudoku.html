089        /**
090         * @return Crea un cuadro completando cada una de las posiciones libres con un valor todavía no asignado
091         */
092        public static CuadroSudoku createCompletadoPorPosicionesLibres() {
093                Multiset<Integer> ms = HashMultiset.create();
094                ms.addAll(CuadroSudoku.casillasOcupadas.values());
095                List<Integer> valoresEnPosicionesLibres = IntStream.rangeClosed(1,CuadroSudoku.numeroDeFilas)
096                                .boxed()
097                                .map(x->Lists2.nCopias(CuadroSudoku.numeroDeFilas-ms.count(x),x).stream())
098                                .flatMap(x->x)
099                                .collect(Collectors.toList());
100                return new CuadroSudoku(valoresEnPosicionesLibres);
101        }

103        /**
104         * @return Crea un cuadro completando las casillas libres de cada fila con los valores no fijados en la misma
105         */
106        public static CuadroSudoku createCompletadoPorFilas() {
107                CuadroSudoku c = CuadroSudoku.create();
108                List<Integer> valoresEnPosicionesLibres = Lists.newArrayList();
109                CuadroSudoku.comienzoDeFilas = Lists.newArrayList();
110                Set<Integer> s = IntStream.rangeClosed(1, CuadroSudoku.numeroDeFilas).boxed().collect(Collectors.toSet());
111                int n = 0;
112                for (int i = 0; i < CuadroSudoku.numeroDeFilas; i++) {
113                        CuadroSudoku.comienzoDeFilas.add(n);
114                        Set<Integer> libres = Sets.newHashSet(s);
115                        libres.removeAll(c.valoresOcupadosEnY.get(i));
116                        valoresEnPosicionesLibres.addAll(libres);
117                        n = n + libres.size();
118                }
119                return new CuadroSudoku(valoresEnPosicionesLibres);
120        }

161        /**
162         * Asigna valores a las casillas que sólo pueden tener un valor y reduce el número de casillas libres
163         */
164        public static void asignaValoresUnicos() {
165                CuadroSudoku.sortPosicionesLibresPorNumVal();
166                List<List<Integer>> lv = CuadroSudoku.valoresLibres();
167                List<List<Integer>> lv2 = lv.stream().filter(x->x.size()==1).collect(Collectors.toList());
168                int n = lv2.size();
169                while (n>0) {
170                        for (int i = 0; i < n; i++) {
171                                Casilla c = Casilla.create(CuadroSudoku.posicionesLibres.get(i));
172                                CuadroSudoku.casillasOcupadas.put(c.x, c.y, lv.get(i).get(0));                          
173                        }       
174                        CuadroSudoku.iniLibresYOcupadas();
175                        CuadroSudoku.sortPosicionesLibresPorNumVal();
176                        lv = CuadroSudoku.valoresLibres();
177                        lv2 = lv.stream().filter(x -> x.size() == 1).collect(Collectors.toList());
178                        n = lv2.size();
179                }
180        }

182        private CuadroSudoku(List<Integer> valoresEnPosicionesLibres) {
183                super();
184                this.valoresEnPosicionesLibres = Lists.newArrayList(valoresEnPosicionesLibres);
185                this.listDatos = Lists.newArrayList(CuadroSudoku.valores);

191         * @param valoresEnPosicionesLibres Valores en las posiciones libres
192         * @pos Rellena las posiciones libres con los valores dados
193         */
194        private void completaValores(List<Integer> valoresEnPosicionesLibres) {         
195                for (int i = 0; i < valoresEnPosicionesLibres.size(); i++) {
196                        int p = CuadroSudoku.posicionesLibres.get(i);
197                        int val = valoresEnPosicionesLibres.get(i);             
198                        this.listDatos.set(p,val);
199                }               
200        }

239        private void calculaPropiedadesDerivadas(){
240                
241                this.valoresOcupadosEnX = Lists.newArrayList();
242                this.valoresOcupadosEnY = Lists.newArrayList();
243                this.valoresOcupadosEnSubCuadro = Lists.newArrayList();
244                
245                for(Integer i=0; i< CuadroSudoku.numeroDeFilas;i++){
246                        this.valoresOcupadosEnX.add(Sets.<Integer>newHashSet());
247                        this.valoresOcupadosEnY.add(Sets.<Integer>newHashSet());
248                        this.valoresOcupadosEnSubCuadro.add(Sets.<Integer>newHashSet());
249                }       
250                
251                for (int p = 0; p < CuadroSudoku.numeroDeCasillas; p++) {
252                        Casilla c = Casilla.create(p);
253                        Integer val = this.listDatos.get(p);
254                        this.valoresOcupadosEnX.get(c.x).add(val);
255                        this.valoresOcupadosEnY.get(c.y).add(val);
256                        this.valoresOcupadosEnSubCuadro.get(c.sc).add(val);
257                }
258                
259                this.objetivo = 0;
260                
261                for(int i=0;i<CuadroSudoku.numeroDeFilas;i++){
262                        this.valoresOcupadosEnX.get(i).remove(0);
263                        this.objetivo = this.objetivo + this.valoresOcupadosEnX.get(i).size();
264                        this.valoresOcupadosEnY.get(i).remove(0);
265                        this.objetivo = this.objetivo + this.valoresOcupadosEnY.get(i).size();
266                        this.valoresOcupadosEnSubCuadro.get(i).remove(0);
267                        this.objetivo = this.objetivo + this.valoresOcupadosEnSubCuadro.get(i).size();                  
268                }
269                
270//              this.objetivo = 3*CuadroSudokuAG.numeroDeFilas*CuadroSudokuAG.numeroDeFilas -this.objetivo;             
271        }

275        /**
276         * Inicializa las posiciones libres y ocupadas
277         */
278        public static void iniLibresYOcupadas() {
279                CuadroSudoku.posicionesLibres = Lists.newArrayList();
280                CuadroSudoku.posicionesOcupadas = Lists.newArrayList();
281                CuadroSudoku.posicionesLibresEnfilas = Lists.newArrayList();            
282                for (int i = 0; i < CuadroSudoku.numeroDeFilas; i++) {
283                        CuadroSudoku.posicionesLibresEnfilas.add(Lists.newArrayList());
284                }
285                CuadroSudoku.valores = Lists2.nCopias(CuadroSudoku.numeroDeCasillas, 0);
286                for (int p = 0; p < CuadroSudoku.numeroDeCasillas; p++) {
287                        Casilla c = Casilla.create(p);
288                        if(c.isFree){
289                                CuadroSudoku.posicionesLibres.add(p);
290                                CuadroSudoku.posicionesLibresEnfilas.get(c.y).add(p);
291                        } else {
292                                CuadroSudoku.posicionesOcupadas.add(p);
293                                CuadroSudoku.valores.set(p, CuadroSudoku.casillasOcupadas.get(c.x, c.y));
294                        }                               
295                }       
296                CuadroSudoku.numPosicionesLibres = CuadroSudoku.posicionesLibres.size();
297        }

299        /**
300         * @param p Una posición en el cuadrado
301         * @return El conjunto de valores ocupados en esa posición
302         */
303        public Set<Integer> getValoresOcupadosEnPos(Integer p){
304                Casilla casilla = Casilla.create(p);
305                Set<Integer> ocupados = Sets.newHashSet();
306                ocupados.addAll(this.valoresOcupadosEnX.get(casilla.x));
307                ocupados.addAll(this.valoresOcupadosEnY.get(casilla.y));
308                ocupados.addAll(this.valoresOcupadosEnSubCuadro.get(casilla.sc));
309                return ocupados;
310        }

312        /**
313         * @param p Una posición en el cuadrado
314         * @return El conjunto de valores libres en esa posición
315         */
316        public Set<Integer> getValoresLibresEnPos(Integer p) {
317                if(p>=CuadroSudoku.numeroDeCasillas) return Sets.newHashSet();
318                Set<Integer> s = IntStream.rangeClosed(1,CuadroSudoku.numeroDeFilas)
319                                .boxed().collect(Collectors.toSet());
320                s.removeAll(this.getValoresOcupadosEnPos(p));
321                return s;
322        }

334                CuadroSudoku.iniLibresYOcupadas();

367        @Override
368        public boolean equals(Object obj) {
369                if (this == obj)
370                        return true;
371                if (obj == null)
372                        return false;
373                if (!(obj instanceof CuadroSudoku))
374                        return false;
375                CuadroSudoku other = (CuadroSudoku) obj;
376                if (listDatos == null) {
377                        if (other.listDatos != null)
378                                return false;
379                } else if (!Arrays.deepEquals(listDatos.toArray(),other.listDatos.toArray()))
380                        return false;
381                return true;
382        }
