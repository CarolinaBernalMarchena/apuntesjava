128        /**
129         * @return El número de etiquetas del árbol
130         */
131        public int size(){
132                int r;
133                if(this.isEmpty()){
134                        r = 0;
135                }else {
136                        int sizq = this.left!=null?this.left.size():0;
137                        int sder = this.right!=null?this.right.size():0;
138                        r = sizq+ sder +1;
139                }
140                return r;
141        }

143        /**
144         * @return El número de hijos del árbol
145         */
146        public int getNumChildren() {
147                int r;
148                if(this.left==null && this.right==null){
149                        r = 0;
150                }else if(this.left==null && this.right!=null){
151                        r = 1;
152                }else if(this.left!=null && this.right==null) {
153                        r = 1;
154                } else {
155                        r = 2;
156                }
157                return r;
158        }

161        /**
162         * @return La altura del árbol
163         */
164        public int getHeight(){
165                int r;
166                if(this.isEmpty()){
167                        r = -1;
168                }else {
169                        int hizq = this.left!=null?this.left.getHeight():-1;
170                        int hder = this.right!=null?this.right.getHeight():-1;
171                        r = Math.max(hizq,hder) +1;
172                }
173                return r;
174        }

189                BinaryTree<T> r;
190                if(this.isEmpty()){
191                        r = BinaryTree.create();
192                } else if(this.isLeaf()){
193                        r = BinaryTree.create(this.label);
194                } else {
195                        BinaryTree<T> cizq = this.left!=null?this.left.getCopy():null;
196                        BinaryTree<T> cder = this.right!=null?this.right.getCopy():null;
197                        r = BinaryTree.create(this.label,cizq, cder);
198                }
199                return r;

232        /**
233         * @param t Un árbol
234         * @param <E> el tipo de las etiquetas del árbol
235         * @return Una lista con el recorrido en preorden. Versión iterativa
236         */
237        public static <E> List<E> getPreOrderIter(BinaryTree<E> t){
238                List<E> ls = new ArrayList<>();
239                Stack2<BinaryTree<E>> p = Stack2.create();
240                if(!t.isEmpty()) p.add(t);
241                while(!p.isEmpty()){
242                        BinaryTree<E> a = p.remove();
243                        ls.add(a.getLabel());                   
244                        if(a.right!=null) p.add(a.right);
245                        if(a.left!=null) p.add(a.left);
246                }
247                return ls;
248        }

254        public static <E> List<E> getPreOrderRecur(BinaryTree<E> t){
255                List<E> r; 
256                List<E> emptyList = new ArrayList<>();
257                if(t.isEmpty()){
258                        r = new ArrayList<>();
259                } else if(t.isLeaf()){
260                        r = new ArrayList<>();
261                        r.add(t.getLabel());
262                } else {
263                        List<E> izq = t.left!=null?BinaryTree.getPreOrderRecur(t.left):emptyList;
264                        List<E> der = t.right!=null?BinaryTree.getPreOrderRecur(t.right):emptyList;
265                        r = new ArrayList<>();
266                        r.add(t.getLabel());
267                        r.addAll(izq);
268                        r.addAll(der);
269                }
270                return r;
271        }

285        /**
286         * @param t Un árbol
287         * @param <E> el tipo de las etiquetas del árbol
288         * @return Una lista con el recorrido en postorden. Versión iterativa
289         */
290        public static <E> List<E> getPostOrderIter(BinaryTree<E> t){
291                List<E> ls = new ArrayList<>();
292                Stack2<Entry<BinaryTree<E>,Boolean>> p = Stack2.create();
293                if(!t.isEmpty()) p.add(new SimpleEntry<>(t, false));
294                while(!p.isEmpty()){
295                        Entry<BinaryTree<E>,Boolean> a = p.peek();
296                        if (a.getValue()) {
297                                a = p.remove();
298                                ls.add(a.getKey().getLabel());                          
299                        } else {
300                                a.setValue(true);                               
301                                if (a.getKey().right != null) p.add(new SimpleEntry<>(a.getKey().right,false));
302                                if (a.getKey().left != null) p.add(new SimpleEntry<>(a.getKey().left,false));
303                        }
304                }
305                return ls;
306        }

313                List<E> r; 
314                List<E> emptyList = new ArrayList<>();
315                if(t.isEmpty()){
316                        r = new ArrayList<>();
317                } else if(t.isLeaf()){
318                        r = new ArrayList<>();
319                        r.add(t.getLabel());

321                        List<E> izq = t.left!=null?BinaryTree.getPostOrderRecur(t.left):emptyList;
322                        List<E> der = t.right!=null?BinaryTree.getPostOrderRecur(t.right):emptyList;
323                        r = new ArrayList<>();                  
324                        r.addAll(izq);
325                        r.addAll(der);
326                        r.add(t.getLabel());
327                }
328                return r;
329        }

343         * @param t Un árbol
344         * @param <E> el tipo de las etiquetas del árbol
345         * @return Una lista con el recorrido en inorden. Versión iterativa
346         */
347        public static <E> List<E> getInOrderIter(BinaryTree<E> t){
348                List<E> ls = new ArrayList<>();
349                Stack2<Entry<BinaryTree<E>,Boolean>> p = Stack2.create();
350                if(!t.isEmpty()) p.add(new SimpleEntry<>(t, false));
351                while(!p.isEmpty()){
352                        Entry<BinaryTree<E>,Boolean> a = p.remove();
353                        if (a.getValue()) {
354                                ls.add(a.getKey().getLabel());                          
355                        } else {
356                                a.setValue(true);
357                                if (a.getKey().right != null) p.add(new SimpleEntry<>(a.getKey().right,false));
358                                p.add(a);
359                                if (a.getKey().left != null) p.add(new SimpleEntry<>(a.getKey().left,false));                           
360                        }
361                }
362                return ls;
363        }

400                List<E> ls = new ArrayList<>();
401                Queue2<BinaryTree<E>> p = Queue2.create();
402                if(!t.isEmpty()) p.add(t);
403                while(!p.isEmpty()){
404                        BinaryTree<E> a = p.remove();

412        @Override
413        public int hashCode() {
414                final int prime = 31;
415                int result = 1;
416                result = prime * result + ((label == null) ? 0 : label.hashCode());
417                result = prime * result + ((left == null) ? 0 : left.hashCode());
418                result = prime * result + ((right == null) ? 0 : right.hashCode());
419                return result;
420        }

422        @Override
423        public boolean equals(Object obj) {
424                if (this == obj)
425                        return true;
426                if (obj == null)
427                        return false;
428                if (!(obj instanceof BinaryTree))
429                        return false;
430                BinaryTree<?> other = (BinaryTree<?>) obj;
431                if (label == null) {
432                        if (other.label != null)
433                                return false;
434                } else if (!label.equals(other.label))
435                        return false;
436                if (left == null) {
437                        if (other.left != null)
438                                return false;
439                } else if (!left.equals(other.left))
440                        return false;
441                if (right == null) {
442                        if (other.right != null)
443                                return false;
444                } else if (!right.equals(other.right))
445                        return false;
446                return true;
447        }

449        @Override
450        public String toString() {
451                String s;
452                if(this.isEmpty()){
453                        s = "()";
454                } else if(this.isLeaf()){
455                        s = this.label.toString();
456                } else if(this.left != null && this.right != null){
457                        s = this.label+"("+this.left+","+this.right+")";
458                } else if(this.left == null){
459                        s = this.label+"("+this.right+")";
460                } else {
461                        s = this.label+"("+this.left+")";
462                }       
463                return  s;
464        }
