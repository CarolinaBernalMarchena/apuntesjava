001package us.lsi.common;
002
003import com.google.common.base.Preconditions;
004
005import java.util.List;
006
007public class Pair<T1, T2> {
008
009        public static <T1, T2> Pair<T1, T2> create(T1 p1, T2 p2) {
010                return new Pair<T1, T2>(p1, p2);
011        }
012
013        public static <T1, T2> Pair<T1, T2> create(Pair<T1, T2> t) {
014                return new Pair<T1, T2>(t.getP1(), t.getP2());
015        }
016        
017        public static <T> Pair<T, T> create(List<T> t) {
018                Preconditions.checkArgument(t.size()==2);
019                return create(t.get(0), t.get(1));
020        }
021        
022        public static <T> Pair<T, T> create(T[] t) {
023                Preconditions.checkArgument(t.length==2);
024                return create(t[0], t[1]);
025        }
026        
027        public T1 p1;
028        public T2 p2;
029        
030        protected Pair(T1 p1, T2 p2) {
031                super();
032                this.p1 = p1;
033                this.p2 = p2;
034        }
035        public T1 getP1() {
036                return p1;
037        }
038        public T2 getP2() {
039                return p2;
040        }
041        
042        public Pair<T1,T2> setP1(T1 v){
043                return create(v,getP2());
044        }
045        
046        public Pair<T1,T2> setP2(T2 v){
047                return create(getP1(),v);
048        }
049        
050        public void set(T1 v1, T2 v2){
051                this.p1 = v1;
052                this.p2 = v2;
053        }
054        
055        @Override
056        public String toString() {
057                return "(" + p1 + "," + p2 + ")";
058        }
059        
060        @Override
061        public int hashCode() {
062                final int prime = 31;
063                int result = 1;
064                result = prime * result + ((p1 == null) ? 0 : p1.hashCode());
065                result = prime * result + ((p2 == null) ? 0 : p2.hashCode());
066                return result;
067        }
068        
069        @Override
070        public boolean equals(Object obj) {
071                if (this == obj)
072                        return true;
073                if (obj == null)
074                        return false;
075                if (getClass() != obj.getClass())
076                        return false;
077                Pair<?,?> other = (Pair<?, ?>) obj;
078                if (p1 == null) {
079                        if (other.p1 != null)
080                                return false;
081                } else if (!p1.equals(other.p1))
082                        return false;
083                if (p2 == null) {
084                        if (other.p2 != null)
085                                return false;
086                } else if (!p2.equals(other.p2))
087                        return false;
088                return true;
089        }
090        
091        
092        
093}
