001package us.lsi.common;
002
003import java.util.Arrays;
004import java.util.Collection;
005import java.util.Set;
006
007import us.lsi.stream.Stream2;
008
009import java.util.stream.Collectors;
010import java.util.stream.IntStream;
011
012public class Sets2 {
013
014        public static Set<Integer> newSet(Integer a, Integer b, Integer c){             
015                return Stream2.create(Stream2.range(a, b, c).boxed()).toSet();
016        }       
017        
018        public static Set<Integer> newSet(Integer a, Integer b){
019                return IntStream.range(a,b).boxed().collect(Collectors.toSet());
020        }
021        
022        @SafeVarargs
023        public static <E> Set<E> newSet(E... e){
024                return Arrays.stream(e).collect(Collectors.toSet());
025        }
026
027        public static <E,U extends Collection<E>> Set<E> newList(U elements){
028                return elements.stream().collect(Collectors.toSet());
029        }
030}
