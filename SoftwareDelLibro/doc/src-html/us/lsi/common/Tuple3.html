001package us.lsi.common;
002
003import java.util.List;
004
005import com.google.common.base.Preconditions;
006
007public class Tuple3<T1, T2, T3> {
008
009        public static <T1, T2, T3> Tuple3<T1, T2, T3> create(T1 p1, T2 p2, T3 p3) {
010                return new Tuple3<T1, T2, T3>(p1, p2, p3);
011        }
012        
013        public static <T1, T2, T3> Tuple3<T1, T2, T3> create(Tuple3<T1, T2, T3> t) {
014                return new Tuple3<T1, T2, T3>(t.getV1(), t.getV2(), t.getV3());
015        }
016        
017        public static <T> Tuple3<T, T, T> create(List<T> t) {
018                Preconditions.checkArgument(t.size()==3);
019                return create(t.get(0), t.get(1), t.get(2));
020        }
021        
022        public static <T> Tuple3<T, T, T> create(T[] t) {
023                Preconditions.checkArgument(t.length==3);
024                return create(t[0], t[1], t[2]);
025        }
026        
027        
028        private final T1 v1;
029        private final T2 v2;
030        private final T3 v3;
031        
032        private Tuple3(T1 p1, T2 p2, T3 p3) {
033                super();
034                this.v1 = p1;
035                this.v2 = p2;
036                this.v3 = p3;
037        }       
038        
039        public T1 getV1() {
040                return v1;
041        }
042
043        public T2 getV2() {
044                return v2;
045        }
046
047        public T3 getV3() {
048                return v3;
049        }
050
051        public Tuple3<T1,T2,T3> setP1(T1 v){
052                return create(v,getV2(),getV3());
053        }
054        
055        public Tuple3<T1,T2,T3> setP2(T2 v){
056                return create(getV1(),v,getV3());
057        }
058        
059        public Tuple3<T1,T2,T3> setP3(T3 v){
060                return create(getV1(),getV2(),v);
061        }
062        
063        @Override
064        public String toString() {
065                return "(" + v1 + "," + v2 + "," + v3 + ")";
066        }
067        @Override
068        public int hashCode() {
069                final int prime = 31;
070                int result = 1;
071                result = prime * result + ((v1 == null) ? 0 : v1.hashCode());
072                result = prime * result + ((v2 == null) ? 0 : v2.hashCode());
073                result = prime * result + ((v3 == null) ? 0 : v3.hashCode());
074                return result;
075        }
076        @Override
077        public boolean equals(Object obj) {
078                if (this == obj)
079                        return true;
080                if (obj == null)
081                        return false;
082                if (getClass() != obj.getClass())
083                        return false;
084                Tuple3<?, ?, ?> other = (Tuple3<?, ?, ?>) obj;
085                if (v1 == null) {
086                        if (other.v1 != null)
087                                return false;
088                } else if (!v1.equals(other.v1))
089                        return false;
090                if (v2 == null) {
091                        if (other.v2 != null)
092                                return false;
093                } else if (!v2.equals(other.v2))
094                        return false;
095                if (v3 == null) {
096                        if (other.v3 != null)
097                                return false;
098                } else if (!v3.equals(other.v3))
099                        return false;
100                return true;
101        }
102        
103        
104}
