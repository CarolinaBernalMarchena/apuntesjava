001package us.lsi.common;
002
003import java.util.Arrays;
004import java.util.HashMap;
005import java.util.Map;
006
007import com.google.common.collect.HashMultiset;
008import com.google.common.collect.Multiset;
009
010public class Multisets2 {
011
012        /**
013         * @param <E> Tipo de los elementos del Multiset
014         * @return Un Multiset vacío
015         */
016        public static <E> Multiset<E> newHashMultiset(){
017                return HashMultiset.<E>create();
018        }
019
020        /**
021         * @param <E> Tipo de los elementos del Multiset
022         * @param it Un iterable
023         * @return Un Multiset construido a partir del iterable
024         */
025        public static <E> Multiset<E> newHashMultiset(Iterable<E> it){
026                return HashMultiset.<E>create(it);
027        }
028        
029        /**
030         * @param <E> Tipo de los elementos del Multiset
031         * @param m Un Map
032         * @return Un Multiset construido a partir del Map
033         */
034        public static <E> Multiset<E> newHashMultiset(Map<E,Integer> m){
035                Multiset<E> r = HashMultiset.create();
036                m.entrySet().stream().forEach(x->r.add(x.getKey(),x.getValue()));
037                return r;
038        }
039        
040        /**
041         * @param <E> Tipo de los elementos del Multiset
042         * @param m Un Multiset
043         * @return Un Map construido a partir del Multiset
044         */
045        public static <E> Map<E,Integer> asMap(Multiset<E> m){
046                Map<E,Integer> r = new HashMap<>();;
047                m.entrySet().stream().forEach(x->r.put(x.getElement(),x.getCount()));
048                return r;
049        }
050        
051        /**
052         * @param <E> Tipo de los elementos del Multiset
053         * @param entries Una secuencia de elementos
054         * @return Un Multiset construidom a partir de la secuencia de elementos
055         */
056        public static <E> Multiset<E> newHashMultiset(@SuppressWarnings("unchecked") E... entries){
057                Multiset<E> s = HashMultiset.<E>create();
058                Arrays.asList(entries)
059                .stream()
060                .forEach(x->s.add(x));          
061                return s;
062        }
063        
064}
