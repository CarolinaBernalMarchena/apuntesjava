001package us.lsi.common;
002
003import java.util.HashMap;
004import java.util.Map;
005
006
007
008
009public class Colors {
010        
011        private static Map<Integer, String> mapColors = null;
012                
013        public static String getNameOfColor(Integer color) {                    
014                if (Colors.mapColors==null) {
015                        Map<Integer,String> m;
016                        m = new HashMap<>();
017                        m.put(0, "green");
018                        m.put(1, "red");
019                        m.put(2, "yellow");
020                        m.put(3, "gray");
021                        m.put(4, "cyan");
022                        m.put(5, "orange");
023                        m.put(5, "magenta");
024                        Colors.mapColors = m;
025                }
026                return Colors.mapColors.get(color);
027        }
028        
029
030}
