058            throw new IllegalArgumentException("Graph cannot be null!");

063        }

067        this.goal = null;

125    public GraphPath<V, E> getShortestPath(){
126        this.initialize();
127        gScoreMap.put(sourceVertex, graph.getVertexWeight(sourceVertex));
128        FibonacciHeapNode<V> heapNode = new FibonacciHeapNode<>(sourceVertex);
129        openList.insert(heapNode, graph.getVertexWeight(sourceVertex));
130        vertexToHeapNodeMap.put(sourceVertex, heapNode);
131
132        do {
133            FibonacciHeapNode<V> currentNode = openList.removeMin();
134
135                        // Check whether we reached the target vertex
136                        if ((goal != null && goal.test(currentNode.getData()) || 
137                                 currentNode.getData().equals(targetVertex))
138                                ) {
139                                // Build the path
140                                // return this.buildGraphPath(sourceVertex, targetVertex,
141                                // currentNode.getKey());
142                                E edgeBeforeCurrent = cameFrom.get(currentNode.getData());
143                                return this.buildGraphPath(sourceVertex, currentNode.getData(),         
144                                                currentNode.getKey()+graph.getVertexWeight(currentNode.getData(), edgeBeforeCurrent, null));
145                        }
146
147            // We haven't reached the target vertex yet; expand the node
148            expandNode(currentNode, targetVertex);
149            closedList.add(currentNode.getData());
150        } while (!openList.isEmpty());
151
152        // No path exists from sourceVertex to TargetVertex
153        return null;
154    }

156    @SuppressWarnings("unchecked")
157        private void expandNode(FibonacciHeapNode<V> currentNode, V endVertex)
158    {
159        numberOfExpandedNodes++;
160
161        Set<E> outgoingEdges = null;
162        if (graph instanceof UndirectedGraph) {
163            outgoingEdges = graph.edgesOf(currentNode.getData());
164        } else if (graph instanceof DirectedGraph) {
165            outgoingEdges = ((DirectedGraph<V, E>) graph).outgoingEdgesOf(currentNode.getData());
166        }
167
168        
169        for (E edge : outgoingEdges) {
170            V successor = Graphs.getOppositeVertex(graph, edge, currentNode.getData());
171            if ((successor == currentNode.getData()) || closedList.contains(successor)) { // Ignore
172                                                                                          // self-loops
173                                                                                          // or
174                                                                                          // nodes
175                                                                                          // which
176                                                                                          // have
177                                                                                          // already
178                                                                                          // been
179                                                                                          // expanded
180                continue;
181            }
182            
183            E edgeBeforeCurrent = cameFrom.get(currentNode.getData());
184            double gScore_current = gScoreMap.get(currentNode.getData());
185            double tentativeGScore = gScore_current + 
186                                         graph.getEdgeWeight(edge)+
187                                         graph.getVertexWeight(successor)+
188                                                         graph.getVertexWeight(currentNode.getData(), edgeBeforeCurrent, edge);
189
190            if (!vertexToHeapNodeMap.containsKey(successor)
191                || (tentativeGScore < gScoreMap.get(successor)))
192            {
193                cameFrom.put(successor, edge);
194                gScoreMap.put(successor, tentativeGScore);
195
196                double fScore =
197                    tentativeGScore + graph.getWeightToEnd(successor, endVertex);
198                if (!vertexToHeapNodeMap.containsKey(successor)) {
199                    FibonacciHeapNode<V> heapNode = new FibonacciHeapNode<>(successor);
200                    openList.insert(heapNode, fScore);
201                    vertexToHeapNodeMap.put(successor, heapNode);
202                } else {
203                    openList.decreaseKey(vertexToHeapNodeMap.get(successor), fScore);
204                }
205            }
206        }
207    }

209    /**
210     * Builds the graph path
211     *
212     * @param startVertex starting vertex of the path
213     * @param targetVertex ending vertex of the path
214     * @param pathLength length of the path
215     *
216     * @return the shortest path from startVertex to endVertex
217     */
218    private GraphPath<V, E> buildGraphPath(V startVertex, V targetVertex, double pathLength)
219    {
220        List<E> edgeList = new ArrayList<>();
221        List<V> vertexList = new ArrayList<>();
222        vertexList.add(targetVertex);
223
224        V v = targetVertex;
225      while (!v.equals(startVertex)) {  //cambiado por mí
226//    while (v != startVertex) {
227            edgeList.add(cameFrom.get(v));
228            v = Graphs.getOppositeVertex(graph, cameFrom.get(v), v);
229            vertexList.add(v);
230        }
231        Collections.reverse(edgeList);
232        Collections.reverse(vertexList);
233        return new GraphWalk<>(graph, startVertex, targetVertex,edgeList, pathLength);
234    }
