001package us.lsi.astar;
002
003import java.util.List;
004import java.util.function.Predicate;
005
006import org.jgrapht.GraphPath;
007
008/**
009 * Interface y factoría de un algoritmo A*. 
010 * 
011 * @author Miguel Toro
012 *
013 * @param <V> Tipo de los vértices
014 * @param <E> Tipo de las aristas
015 */
016public interface AStarAlgorithm<V, E> {
017        
018        
019        public static <V,E> AStarAlgorithm<V,E> create(AStarGraph<V, E> graph,V startVertex, V endVertex){
020                return new AStarAlgorithm2<>(graph,startVertex,endVertex);
021        }
022        
023        public static <V,E> AStarAlgorithm<V,E> create(AStarGraph<V, E> graph,V startVertex, Predicate<V> goal){
024                return new AStarAlgorithm2<>(graph,startVertex,goal);
025        }
026        
027        public GraphPath<V, E> getPath();
028        public List<V> getPathVertexList();
029        
030        
031}
