001package us.lsi.astar.laberinto;
002
003import java.util.List;
004import java.util.Set;
005import java.util.stream.Collectors;
006import java.util.stream.Stream;
007
008import org.jgrapht.GraphPath;
009
010
011import com.google.common.collect.HashBasedTable;
012import com.google.common.collect.Lists;
013import com.google.common.collect.Table;
014
015import us.lsi.common.PairInteger;
016import us.lsi.graphs.*;
017import us.lsi.stream.Stream2;
018
019public class Casilla implements VirtualVertex<Casilla,SimpleEdge<Casilla>> {
020        private Integer x;
021        private Integer y;
022        private Integer info;
023        public static int numX;
024        public static int numY;
025        public static Table<Integer,Integer,Integer> datosCasillas = HashBasedTable.<Integer,Integer,Integer>create();
026        
027        private Casilla(int x, int y) {
028                super();
029                this.x = x;
030                this.y = y;
031                this.info = datosCasillas.get(y, y);
032                if(!isValid()){
033                        throw new IllegalArgumentException("Casilla no valida con "+this.x+","+this.y+","+this.info);
034                }
035        }
036        
037        public Integer getX() {
038                return x;
039        }
040        public void setX(int x) {
041                this.x = x;
042        }
043        public Integer getY() {
044                return y;
045        }
046        public void setY(int y) {
047                this.y = y;
048        }
049        
050        public Integer getInfo() {
051                return info;
052        }
053
054        public static void iniDatos(String nf, int nx, int ny){
055                numX = nx;
056                numY = ny;
057                Stream<Integer> ls = Stream2.fromFile(nf)
058                                .<String>flatMap((String x)->Stream2.fromString(x, ","))
059                                .<Integer>map((String x)->new Integer(x));
060                
061                
062                Container ct = new Container(Casilla.datosCasillas);
063                
064                ls.forEach((Integer e)->{ ct.add(e); });
065                                
066                if(ct.numDatos!= numX*numY){
067                        throw new IllegalArgumentException("No hay el número adecuado de datos");
068                }
069        }
070
071        
072        public static Casilla create(int x, int y) {
073                return new Casilla(x, y);
074        }
075
076        @Override
077        public int hashCode() {
078                final int prime = 31;
079                int result = 1;
080                result = prime * result + x;
081                result = prime * result + y;
082                return result;
083        }
084        @Override
085        public boolean equals(Object obj) {
086                if (this == obj)
087                        return true;
088                if (obj == null)
089                        return false;
090                if (getClass() != obj.getClass())
091                        return false;
092                Casilla other = (Casilla) obj;
093                if (x != other.x)
094                        return false;
095                if (y != other.y)
096                        return false;
097                return true;
098        }
099
100        @Override
101        public String toString() {
102                return "(" + x + "," + y + "," + info + ")";
103        }
104
105        @Override
106        public Set<Casilla> getNeighborListOf() {
107                List<PairInteger> ls = Lists.newArrayList(PairInteger.create(1,0),
108                                PairInteger.create(0,1),PairInteger.create(-1,0),PairInteger.create(0,-1));
109                return ls.stream()
110                                 .<PairInteger> map((PairInteger x)-> PairInteger.create(x.v1+this.x, x.v2+this.y))
111                                 .filter((PairInteger x)-> x.v1>=0 && x.v1 < numX && x.v2>=0 && x.v2 < numY && 
112                                                  datosCasillas.get(x.v1, x.v2) >=0)
113                                 .<Casilla>map((PairInteger x)-> Casilla.create(x.v1,x.v2))
114                                 .collect(Collectors.<Casilla>toSet());
115        }
116
117        @Override
118        public Set<SimpleEdge<Casilla>> edgesOf() {
119                return getNeighborListOf().stream()
120                                .map(c->SimpleEdge.create(this,c))
121                                .collect(Collectors.toSet());
122        }
123
124        @Override
125        public boolean isNeighbor(Casilla e) {
126                return Math.abs(this.getX()-e.getX()) + Math.abs(this.getY()-e.getY()) == 1;
127        }
128        
129        @Override
130        public boolean isValid() {
131                return this.info >=0 && this.getX() >=0 && this.getX()< numX && this.getY() >=0 && this.getY()< numY;
132        }
133        
134        private static String normalize(String s, int n){
135                int t = n-s.length();
136                String r = "";
137                for(int i =0;i<t;i++){
138                        r= r+" ";
139                }       
140                r = r+s;
141                return r;
142        }
143        
144        public static String show(GraphPath<Casilla,SimpleEdge<Casilla>> p){
145                List<Casilla> lc = p.getVertexList();           
146                String s = "";
147                Casilla c;
148                for(int y = Casilla.numY-1; y>=0; y--){
149                        for(int x=0; x < Casilla.numX;x++){
150                                        c = Casilla.create(x, y);
151                                        if(lc.contains(c)){
152                                                s = s+normalize("X",7);
153                                        }else{
154                                                s = s+normalize(Casilla.datosCasillas.get(x, y).toString(),7);
155                                        }
156                        }
157                        s = s+"\n";
158                }
159            return s;
160        }
161        
162        private static class Container {
163                private Table<Integer,Integer,Integer> datosCasillas;
164                private int ax = 0;
165                private int ay = numY-1;
166                public int numDatos;
167                        
168                public Container(Table<Integer, Integer, Integer> datosCasillas) {
169                        super();
170                        this.datosCasillas = datosCasillas;
171                        this.ax = 0;
172                        this.ay = numY-1;
173                        this.numDatos = 0;
174                }
175
176
177                public void add(Integer e){
178                        this.datosCasillas.put(ax, ay, e) ;
179                        numDatos++;
180                        if(ax==Casilla.numX){
181                                ax=0;
182                                ay--;
183                        }else{
184                                ax++;
185                        }
186                }
187        }
