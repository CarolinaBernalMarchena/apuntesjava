001package us.lsi.astar.puzzle2;
002
003import java.util.*;
004
005import com.google.common.collect.Lists;
006
007import us.lsi.common.*;
008import us.lsi.graphs.AlternativeVirtualVertex;
009
010/**
011 * 
012 * * Estado de un Puzzle como implementación de AlternativeVirtualVertex. Esta versión simplifica el código con repecto 
013 * a la a laversión que implementa VirtualVertex directamente.
014 * 
015 * @author Miguel Toro
016 *
017 */
018public class EstadoPuzzle extends AlternativeVirtualVertex<PairInteger,EstadoPuzzle> {
019        
020        /**
021         * @param d Lista de valores del puzzle dados por filas de abajo arriba
022         * @return Un EstadoPuzzle
023         */
024        public static EstadoPuzzle create(Integer... d) {
025                return new EstadoPuzzle(d);
026        }
027        
028        private static List<PairInteger> values = null;
029        
030        private Integer[][] datos; 
031        private int i0;
032        private int j0;
033        public static int numFilas = 3;
034        
035        private EstadoPuzzle(Integer... d) {
036                super(); 
037                this.datos = new Integer[EstadoPuzzle.numFilas][EstadoPuzzle.numFilas];
038                Set<Integer> s = new HashSet<Integer>();        
039                int i=0,j=0;
040                for (int e:d) {                                                 
041                                        if(e<0 || e > (EstadoPuzzle.numFilas*EstadoPuzzle.numFilas-1)){
042                                                throw new IllegalArgumentException();
043                                        }
044                                        if(e==0){
045                                                this.i0 =i;
046                                                this.j0 =j;
047                                        }
048                                        s.add(e);
049                                        this.datos[i][j] = e;
050                                        if(i== EstadoPuzzle.numFilas-1){
051                                                j++;
052                                                i=0;
053                                        }else{
054                                                i++;
055                                        }
056                                
057                }
058                if(d.length!= EstadoPuzzle.numFilas*EstadoPuzzle.numFilas || s.size()!= EstadoPuzzle.numFilas*EstadoPuzzle.numFilas)
059                        throw new IllegalArgumentException();
060        }
061
062
063        public EstadoPuzzle(Integer[][] datos, int i0, int j0) {
064                super();
065                this.datos = datos;
066                this.i0 = i0;
067                this.j0 = j0;
068        }
069
070        public Integer getDato(int i, int j) {
071                if(i<0 || i>=EstadoPuzzle.numFilas || j<0 || j>=EstadoPuzzle.numFilas)
072                        throw new IllegalArgumentException();
073                return datos[i][j];
074        }
075        
076        @Override
077        public boolean isNeighbor(EstadoPuzzle e) {
078                return this.getNumDiferentes(e)==2 && Math.abs(this.i0-e.i0)+Math.abs(this.j0-e.j0)==1;
079        }
080        
081        @Override
082        public boolean isValid() {
083                Set<Integer> s = new HashSet<Integer>();        
084                for (int j = 0; j < EstadoPuzzle.numFilas; j++) {
085                        for (int i = 0; i < EstadoPuzzle.numFilas; i++) {
086                                s.add(datos[i][j]);
087                        }
088                }
089                return s.size()== EstadoPuzzle.numFilas*EstadoPuzzle.numFilas;
090        }
091        
092        public Integer getNumDiferentes(EstadoPuzzle e){
093                Integer s = 0;
094                for (int i = 0; i < EstadoPuzzle.numFilas; i++) {
095                        for (int j = 0; j < EstadoPuzzle.numFilas; j++) {
096                                if (!this.datos[i][j].equals(e.datos[i][j])) {
097                                        s++;
098                                }
099                        }
100                }
101                return s;
102        }
103        
104        @Override
105        public int hashCode() {
106                final int prime = 31;
107                int result = 1;
108                result = prime * result + Arrays.deepHashCode(datos);
109                return result;
110        }
111
112
113        @Override
114        public boolean equals(Object obj) {
115                if (this == obj)
116                        return true;
117                if (obj == null)
118                        return false;
119                if (!(obj instanceof EstadoPuzzle))
120                        return false;
121                EstadoPuzzle other = (EstadoPuzzle) obj;
122                if (!Arrays.deepEquals(datos, other.datos))
123                        return false;
124                return true;
125        }
126
127
128        @Override
129        public String toString() {
130                String s ="\n";
131                
132                for (int j = datos.length-1; j >= 0; j--) {
133                        for (int i = 0; i < datos.length; i++) {
134                                if (i<datos.length-1) {
135                                        s = s + datos[i][j]+",";
136                                }else {
137                                        s = s + datos[i][j]+"\n";
138                                }
139                        }
140                }
141                return s;
142        }
143
144
145        @Override
146        public EstadoPuzzle neighbor(PairInteger p) {
147                if(!isApplicable(p))
148                        throw new IllegalArgumentException();
149                int i0 = this.i0 + p.p1;
150                int j0 = this.j0 + p.p2;
151                Integer[][] dd = new Integer[EstadoPuzzle.numFilas][EstadoPuzzle.numFilas];
152                for (int i = 0; i < EstadoPuzzle.numFilas; i++) {
153                        for (int j = 0; j < EstadoPuzzle.numFilas; j++) {
154                                dd[i][j] = datos[i][j];
155                        }
156                }
157                dd[i0][j0] = datos[this.i0][this.j0];
158                dd[this.i0][this.j0] = datos[i0][j0];
159                return new EstadoPuzzle(dd,i0,j0);
160        }
161
162
163        @Override
164        public boolean isApplicable(PairInteger p) {
165                return p.p1+this.i0>=0 && p.p1+this.i0<EstadoPuzzle.numFilas && 
166                                p.p2+this.j0>=0 && p.p2+this.j0<EstadoPuzzle.numFilas;
167        }
168
169
170        @Override
171        protected List<PairInteger> values() {
172                if(EstadoPuzzle.values==null){
173                        EstadoPuzzle.values = Lists.newArrayList(PairInteger.create(1,0),PairInteger.create(0,1),PairInteger.create(-1,0),PairInteger.create(0,-1));
174                }
175                return EstadoPuzzle.values;
176        }
177
178
179        @Override
180        protected EstadoPuzzle getThis() {
181                return this;
182        }
183        
184        
185}
