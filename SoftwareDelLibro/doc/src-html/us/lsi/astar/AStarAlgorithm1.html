004        
005import java.util.*;
006import java.util.function.Predicate;
007
008import org.jgrapht.*;
009import org.jgrapht.graph.*;

013/**
014 * <p> Implementación del algoritmo A*. Adaptación de la clase DijkstraShortestPath en <p> <a href="http://jgrapht.org/javadoc/" target="_blank">JGrapht</a></p>
015 * 
016 * @author Miguel Toro
017 *
018 * @param <V> Tipo de los vértices
019 * @param <E> Tipo de las aristas
020 */
021public final class AStarAlgorithm1<V, E> implements AStarAlgorithm<V, E> {
022        
023
024        private GraphPath<V, E> path;
025        private AStarIterator<V, E> iterator;
026        private AStarGraph<V, E> graph;
027        private V startVertex;
028        private V endVertex;
029        private Predicate<V> goal=null;
030        
031        /**
032         * <p> El algoritmo para alcanzar el vértice destino </p>
033         * 
034         * @param graph Grafo 
035         * @param startVertex Vértice origen
036         * @param endVertex Vértice destino
037         * 
038         */
039        public AStarAlgorithm1(AStarGraph<V, E> graph,V startVertex, V endVertex) {
040                this(graph, startVertex, endVertex, null,
041                                Double.POSITIVE_INFINITY);
042        }
043
044        /**
045         * <p> El algoritmo para alcanzae el primer vértice cuya distancia al objetivo es cero </p>
046         * 
047         * @param graph Grafo
048         * @param startVertex Vértice origen
049         * @param goal Predicado que indica las condiciones de un vértice final
050         * 
051         */
052        public AStarAlgorithm1(AStarGraph<V, E> graph, V startVertex, Predicate<V> goal) {
053                this(graph, startVertex, null, goal,
054                                Double.POSITIVE_INFINITY);
055        }
056
057        private AStarAlgorithm1(AStarGraph<V, E> graph, V startVertex, V endVertex, Predicate<V> goal, double radius) {
058
059                
060                this.graph=graph;
061                this.startVertex = startVertex;
062                this.endVertex = endVertex;
063                this.goal = goal;
064                
065                
066                if (endVertex!=null && !graph.containsVertex(endVertex)) {
067                        throw new IllegalArgumentException(
068                                        "graph must contain the end vertex");
069                }
070
071                iterator = new AStarIterator<V, E>(graph, startVertex, endVertex, radius);
072                
073                while (iterator.hasNext()) {            
074                        
075                        V vertex = iterator.next();
076                        
077                        
078                        if (this.endVertex != null && vertex.equals(this.endVertex)
079                            || (this.goal != null && this.goal.test(vertex))) {
080                                
081                                path = createEdgeList(graph, iterator, startVertex, vertex);
082                                return;
083                        }
084                        
085                        
086                }
087
088                path = null;
089        }
090
091        /**
092         * @return Devuelve una lista con las aristas del camino mínimo desde el origen al último vértice alcanzado
093         */
094        public List<E> getPathEdgeList() {
095                if (path == null) {
096                        return null;
097                } else {
098                        return path.getEdgeList();
099                }
100        }
101        /**
102         * @return Devuelve el camino mínimo desde el origen al último vértice alcanzado
103         */
104        public GraphPath<V, E> getPath() {
105                return path;
106        }
107        
108        /**
109         * @param vertex Vértice destino
110         * @return Devuelve el camino mínimo desde el origen a vertex
111         */
112        public GraphPath<V, E> getPath(V vertex) {
113                return createEdgeList(graph,iterator,startVertex,vertex);
114        }
115
116        /**
117         * @return Devuelve el peso del camino mínimo desde el origen al último vértice alcanzado
118         */
119        public double getPathLength() {
120                if (path == null) {
121                        return Double.POSITIVE_INFINITY;
122                } else {
123                        return path.getWeight();
124                }
125        }
126        
127        /**
128         * @return Devuelve el iterador que se ha creado para recorrer el grafo
129         */
130        public AStarIterator<V, E> getIterator() {
131                return iterator;
132        }
133        
134        /**
135         * @param graph Grafo
136         * @param startVertex Vértice origen
137         * @param endVertex Vértice Destino
138         * @param <V> El tipo de los elementos vértices
139         * @param <E> El tipo de los elementos de las aristas
140         * @return Devuelve el camino mínimo desde el origen vértice destino
141         */
142        public static <V, E> List<E> findPathBetween(
143                        AStarGraph<V, E> graph, V startVertex, V endVertex) {
144
145                AStarAlgorithm1<V, E> alg = new AStarAlgorithm1<>(graph, startVertex, endVertex);
146
147                return alg.getPathEdgeList();
148        }
149
150        private GraphPath<V, E> createEdgeList(Graph<V, E> graph,
151                        AStarIterator<V, E> iter, V startVertex, V endVertex) {
152
153                List<E> edgeList = new ArrayList<E>();
154
155                V v = endVertex;
156
157                while (true) {
158                        E edge = iter.getSpanningTreeEdge(v);
159
160                        if (edge == null) {
161                                break;
162                        }
163
164                        edgeList.add(edge);
165                        v = Graphs.getOppositeVertex(graph, edge, v);
166                }
167
168                Collections.reverse(edgeList);
169                double pathLength = iter.getShortestPathLength(endVertex);
170                return new GraphWalk<V, E>(graph, startVertex, endVertex, edgeList,
171                                pathLength);
172        }
173
174        @Override
175        public List<V> getPathVertexList() {
176                return getPath().getVertexList();
177        }
178}
