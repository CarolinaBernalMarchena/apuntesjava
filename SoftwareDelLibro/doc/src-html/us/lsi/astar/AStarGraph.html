001package us.lsi.astar;
002
003
004import org.jgrapht.Graph;
005
006public interface AStarGraph<V, E> extends Graph<V,E> {
007        
008        /**
009         * @param edge es una arista
010         * @return El peso de edge
011         */
012        default double getEdgeWeight(E edge) {
013                return 1.;
014        }
015        /**
016         * @param vertex es el vértice actual
017         * @return El peso de vertex
018         */
019        default double getVertexWeight(V vertex) {
020                return 0.;
021        }
022        /**
023         * @param vertex El vértice actual
024         * @param edgeIn Una arista entrante o incidente en el vértice actual. Es null en el vértice inicial.
025         * @param edgeOut Una arista saliente o incidente en el vértice actual. Es null en el vértice final.
026         * @return El peso asociado al vértice suponiendo las dos aristas dadas. 
027         */
028        default double getVertexWeight(V vertex, E edgeIn, E edgeOut) {
029                return 0.;
030        }
031        
032        /**
033         * @param actual El vértice actual
034         * @param endVertex El vértice destino. Este vértice puede ser null. 
035         * 
036         * @return Una cota inferior del peso del camino desde el vértice actual al destino, 
037         * o desde el vértice actual al conjunto de vértices descrito por un predicado objetivo que se especificará en el AStarAlgorithm.
038         * Debe cumplirse la distancia es cero si el vértice actual cumple el predicado objetivo
039         */
040        default double getWeightToEnd(V actual, V endVertex) {
041                return 0.;
042        }
043        
044        
045        
046}
