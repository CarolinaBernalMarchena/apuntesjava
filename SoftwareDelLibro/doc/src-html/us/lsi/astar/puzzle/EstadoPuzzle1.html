001package us.lsi.astar.puzzle;
002
003import java.util.*;
004import java.util.stream.Collectors;
005
006
007

013import com.google.common.collect.Lists;
014
015import us.lsi.common.*;
016import us.lsi.graphs.SimpleEdge;
017import us.lsi.graphs.VirtualVertex;
018
019/**
020 * Estado de un Puzzle como implementación de VirtualVertex
021 * 
022 * 
023 * @author Miguel Toro
024 *
025 */
026public class EstadoPuzzle1 implements VirtualVertex<EstadoPuzzle1,SimpleEdge<EstadoPuzzle1>> {
027        
028        public static EstadoPuzzle1 create(ProblemaPuzzle problema) {
029                return new EstadoPuzzle1(problema);
030        }
031
032        public static EstadoPuzzle1 create(Integer... d) {
033                return new EstadoPuzzle1(ProblemaPuzzle.create(d));
034        }
035        
036        public static EstadoPuzzle1 create(Integer[][] datos, int i0, int j0) {
037                return new EstadoPuzzle1(ProblemaPuzzle.create(datos,i0,j0));
038        }
039        
040        private ProblemaPuzzle problema;
041                
042        
043        private EstadoPuzzle1(ProblemaPuzzle problema) {
044                super();
045                this.problema = problema;
046        }
047
048        
049        public ProblemaPuzzle getProblema() {
050                return problema;
051        }
052
053
054        @Override
055        public boolean isValid() {
056                return problema.isValid();
057        }
058        
059        @Override
060        public Set<EstadoPuzzle1> getNeighborListOf() {
061                List<PairInteger> ls = Lists.newArrayList(PairInteger.create(1, 0),
062                                PairInteger.create(0, 1), PairInteger.create(-1, 0),
063                                PairInteger.create(0, -1));
064                return ls
065                                .stream()
066                                .filter(x -> x.v1 + problema.getI0() >= 0
067                                                && x.v1 + problema.getI0() < ProblemaPuzzle.numFilas
068                                                && x.v2 + problema.getJ0() >= 0
069                                                && x.v2 + problema.getJ0() < ProblemaPuzzle.numFilas)
070                                .map(x -> EstadoPuzzle1.create(problema.getVecino(x.v1, x.v2)))
071                                .collect(Collectors.<EstadoPuzzle1> toSet());
072        }
073        
074        public Set<SimpleEdge<EstadoPuzzle1>> edgesOf(){
075                return this.getNeighborListOf().stream()
076                   .map(x->SimpleEdge.create(this,x))
077                   .collect(Collectors.<SimpleEdge<EstadoPuzzle1>>toSet());
078        }
079
080        Integer getNumDiferentes(EstadoPuzzle1 e){
081                return problema.getNumDiferentes(e.problema);
082        }
083
084        @Override
085        public String toString() {
086                return problema.toString();
087        }
088
089
090        @Override
091        public int hashCode() {
092                final int prime = 31;
093                int result = 1;
094                result = prime * result
095                                + ((problema == null) ? 0 : problema.hashCode());
096                return result;
097        }
098
099
100        @Override
101        public boolean equals(Object obj) {
102                if (this == obj)
103                        return true;
104                if (obj == null)
105                        return false;
106                if (!(obj instanceof EstadoPuzzle1))
107                        return false;
108                EstadoPuzzle1 other = (EstadoPuzzle1) obj;
109                if (problema == null) {
110                        if (other.problema != null)
111                                return false;
112                } else if (!problema.equals(other.problema))
113                        return false;
114                return true;
115        }
116        
117        
118}
