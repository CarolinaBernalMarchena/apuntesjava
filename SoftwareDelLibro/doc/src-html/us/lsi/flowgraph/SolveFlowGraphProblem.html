059        public static FlowGraphSolution solve(
060                        String fileIn, 
061                        String fileGraphFlow, 
062                        String fileSolutionsGraphFlow,
063                        String fileConstraints,
064                        boolean conNombres,
065                        boolean min) {
066                FlowGraph flowGraph = FlowGraph.create(FlowEdge::createEdge);
067                flowGraph = (FlowGraph) GraphsReader.newGraph(fileIn, FlowVertex::create, FlowEdge::createEdge, flowGraph, null);
068                FlowAlgorithm fa = FlowAlgorithm.create(flowGraph, min);
069                ProblemaPL p = fa.getProblemaLP();
070                AlgoritmoPL a = Algoritmos.createPL(p);
071                a.ejecuta();
072                
073                Map<FlowEdge, Double> edgeFlow = fa.getEdgeFlow(a);
074                Map<FlowVertex, Double> sourceFlow = fa.getSourceFlow(a);
075                Map<FlowVertex, Double> sinkFlow = fa.getSinkFlow(a);
076                SolveFlowGraphProblem.saveFileFlow(flowGraph, fileGraphFlow, conNombres);
077                final FlowGraph fgc = flowGraph;
078                
079                FlowGraph sg = Graphs2.<FlowVertex,FlowEdge,FlowGraph>subGraph(
080                                flowGraph, 
081                                v->fgc.edgesOf(v).stream().anyMatch(y->edgeFlow.get(y)>0), 
082                                e->edgeFlow.get(e)>0,
083                                x->FlowGraph.create(x));
084                
085                SolveFlowGraphProblem.saveFileFlowSoutions(sg, fileSolutionsGraphFlow,
086                                sourceFlow, sinkFlow, edgeFlow, conNombres);
087                String constraints = p.toStringConstraints()+fa.toStringIndex();
088                StringExtensions2.toFile(constraints, fileConstraints);
089                Double optFlow = sourceFlow.entrySet().stream().mapToDouble(x->x.getValue()).sum();
090                Double goal = a.getObjetivo();
091                FlowGraphSolution fs = FlowGraphSolution.create(sg, constraints, edgeFlow, sourceFlow, sinkFlow, optFlow, goal);
092                return fs;
093        }

096        public static void saveFileFlow(FlowGraph graph, String file,
097                        boolean conNombres) {
098                GraphsFileExporter.saveFile((Graph<FlowVertex, FlowEdge>) graph, file,
099                                new StringComponentNameProvider<FlowVertex>(), null, null,
100                                new EtiquetasVerticesFlow(graph, conNombres),
101                                new EtiquetasAristasFlow(graph));
102        }
103
104        public static void saveFileFlowSoutions(FlowGraph graph,
105                        String file, Map<FlowVertex, Double> sourceFlow, Map<FlowVertex, Double> sinkFlow,
106                        Map<FlowEdge, Double> edgeFlow, boolean conNombres) {
107
108                GraphsFileExporter.saveFile((Graph<FlowVertex, FlowEdge>) graph, file,
109                                new StringComponentNameProvider<FlowVertex>(), null, null,
110                                new EtiquetasVerticesSolutionFlow<FlowVertex>(sourceFlow, sinkFlow,
111                                                conNombres), new EtiquetasAristasSolutionFlow(
112                                                edgeFlow));
113
114        }

119        private static class EtiquetasVerticesSolutionFlow<V> implements
120                        ComponentAttributeProvider<V> {
121
122                private Map<V, Double> source;
123                private Map<V, Double> sink;
124                private boolean conNombres;
125
126                public EtiquetasVerticesSolutionFlow(Map<V, Double> source,
127                                Map<V, Double> sink, boolean conNombres) {
128                        super();
129                        this.source = source;
130                        this.sink = sink;
131                        this.conNombres = conNombres;
132                }
133
134                @Override
135                public Map<String, String> getComponentAttributes(V v) {
136                        Map<String, String> map = new HashMap<>();
137                        String nombre = conNombres ? v.toString() + "\n" : "";
138                        if (source.containsKey(v)) {
139                                map.put("style", "bold");
140                                map.put("label", nombre + source.get(v).toString());
141                        } else if (sink.containsKey(v)) {
142                                map.put("style", "dotted");
143                                map.put("label", nombre + "-" + sink.get(v));
144                        }
145                        return map;
146                }
147
148        }
149
150        private static class EtiquetasVerticesFlow implements ComponentAttributeProvider<FlowVertex> {
151
152                private FlowGraph graph;
153                boolean conNombres;
154
155                public EtiquetasVerticesFlow(FlowGraph graph, boolean conNombres) {
156                        super();
157                        this.graph = graph;
158                        this.conNombres = conNombres;
159                }
160
161                @Override
162                public Map<String, String> getComponentAttributes(FlowVertex v) {
163                        Map<String, String> map = new HashMap<>();
164                        String nombre = conNombres ? v.toString() + "\n" : "";
165                        String min = graph.getMinVertexWeight(v) > 0 ? ""
166                                        + graph.getMinVertexWeight(v) : "";
167                        String max = graph.getMaxVertexWeight(v) < Double.MAX_VALUE ? ""
168                                        + graph.getMaxVertexWeight(v) : "";
169                        if (min.length() > 0 || max.length() > 0) {
170                                map.put("label", nombre + min + "/" + max);
171                        }
172                        if (graph.isSource(v)) {
173                                map.put("style", "bold");
174                        } else if (graph.isSink(v)) {
175                                map.put("style", "dotted");
176                        }
177                        return map;
178                }
179
180        }

182        private static class EtiquetasAristasSolutionFlow implements ComponentAttributeProvider<FlowEdge> {
183
184                public EtiquetasAristasSolutionFlow(Map<FlowEdge, Double> m) {
185                        super();
186                        this.m = m;
187                }
188
189                private Map<FlowEdge, Double> m;
190
191                @Override
192                public Map<String, String> getComponentAttributes(FlowEdge a) {
193                        Map<String, String> map = new HashMap<>();
194                        if (m.containsKey(a)) {
195                                map.put("label", m.get(a).toString());
196                        }
197                        return map;
198                }

201        private static class EtiquetasAristasFlow implements ComponentAttributeProvider<FlowEdge> {
202
203                public EtiquetasAristasFlow(FlowGraph graph) {
204                        super();
205                        this.graph = graph;
206                }
207
208                private FlowGraph graph;
209
210                @Override
211                public Map<String, String> getComponentAttributes(FlowEdge a) {
212                        Map<String, String> map = new HashMap<>();
213                        String min = graph.getMinEdgeWeight(a) > 0 ? ""
214                                        + graph.getMinEdgeWeight(a) : "";
215                        String max = graph.getEdgeWeight(a) < Double.MAX_VALUE ? ""
216                                        + graph.getEdgeWeight(a) : "";
217                        if (min.length() > 0 || max.length() > 0) {
218                                map.put("label", min + "/" + max);
219                        }
220                        return map;
221                }
