001package us.lsi.bt.reinas;
002
003public class Reina {
004        private Integer y;
005        private Integer x;
006        public static int numeroDeReinas = 8;
007        
008        public static Reina create(int x, int y) {
009                return new Reina(x, y);
010        }
011        
012        Reina(int x, int y) {
013                super();                
014                this.x = x;
015                this.y = y;
016        }
017        
018        public Integer getY() {
019                return y;
020        }
021        public Integer getX() {
022                return x;
023        }
024
025        public Integer getDiagonalPrincipal(){
026                return y-x;
027        }
028        public Integer getDiagonalSecundaria(){
029                return y+x;
030        }
031        
032        @Override
033        public String toString() {
034                return "[" + x + ","+ y + "]";
035        }
036
037        @Override
038        public int hashCode() {
039                final int prime = 31;
040                int result = 1;
041                result = prime * result + ((x == null) ? 0 : x.hashCode());
042                result = prime * result + ((y == null) ? 0 : y.hashCode());
043                return result;
044        }
045
046        @Override
047        public boolean equals(Object obj) {
048                if (this == obj)
049                        return true;
050                if (obj == null)
051                        return false;
052                if (getClass() != obj.getClass())
053                        return false;
054                Reina other = (Reina) obj;
055                if (x == null) {
056                        if (other.x != null)
057                                return false;
058                } else if (!x.equals(other.x))
059                        return false;
060                if (y == null) {
061                        if (other.y != null)
062                                return false;
063                } else if (!y.equals(other.y))
064                        return false;
065                return true;
066        }
067        
068}
