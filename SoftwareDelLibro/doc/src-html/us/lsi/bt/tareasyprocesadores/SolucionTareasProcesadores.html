001package us.lsi.bt.tareasyprocesadores;
002
003import java.util.HashMap;
004import java.util.List;
005import java.util.Map;
006
007import us.lsi.bt.SolucionBT;
008import us.lsi.pd.tareasprocesadores.Tarea;
009
010public class SolucionTareasProcesadores implements SolucionBT {
011
012        public static SolucionTareasProcesadores create(
013                        Map<Integer, List<Tarea>> asignacion, Double objetivo) {
014                return new SolucionTareasProcesadores(asignacion, objetivo);
015        }
016        
017        private Map<Integer,List<Tarea>> asignacion;
018        private Double objetivo;
019
020        private SolucionTareasProcesadores(Map<Integer, List<Tarea>> asignacion,
021                        Double objetivo) {
022                super();
023                this.asignacion = new HashMap<>(asignacion);
024                this.objetivo = objetivo;
025        }
026
027        @Override
028        public Double getObjetivo() {
029                return objetivo;
030        }
031
032        public Map<Integer, List<Tarea>> getAsignacion() {
033                return asignacion;
034        }
035
036
037        @Override
038        public String toString() {
039                return this.objetivo + ","+this.asignacion;
040        }
041        
042        
043}
