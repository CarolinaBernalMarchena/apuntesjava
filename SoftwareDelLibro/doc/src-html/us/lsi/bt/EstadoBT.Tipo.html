001package us.lsi.bt;
002
003import java.util.List;
004
005
006/**
007 * <p> Tipo mutable del estado de un problema que se quiere resolver por la técnica de Bactracking. 
008 * Cada estado está asociado un problema, del conjunto de problemas considerado, más el camino, secuencia de alternativas, para llegar a él desde el problema inicial. </p>
009 * 
010 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema15.pdf" target="_blank">Tema15</a></p>
011 *
012 * @author Miguel Toro
013 *
014 * @param <S> El tipo de la solución
015 * @param <A> El tipo de la alternativa
016 */
017public interface EstadoBT<S extends SolucionBT, A> {
018        
019        
020        public enum Tipo{Min,Max,TodasLasSoluciones,AlgunasSoluciones}
021        
022        /**
023         * @return El tipo del problema: minimiza, maximiza o busca un valor que involucra a todas las alternativas
024         *
025         * 
026         **/
027        Tipo getTipo(); 
028        /**
029         * @return El estado inicial
030         */
031        EstadoBT<S,A> getEstadoInicial();       
032        /**
033         * @param a - alternativa escogida para cambiar el estado
034         * @return El estado siguiente según la alternativa a
035         */
036        EstadoBT<S, A> avanza(A a);
037        /**
038         * @param a - alternativa a eliminar para restaurar el estado anterior
039         * @return El estado anterior
040         */
041        EstadoBT<S, A> retrocede(A a);
042        /**
043         * @return Tamaño del problema al que está asociado el estado
044         */
045        int size();
046        /**
047         * @return Si el problema está fuera de rango y su solución es null
048         */
049        default boolean estaFueraDeRango(){
050                return false;
051        }
052        /**
053         * @return Si el problema asociado al estado es un caso base
054         */
055        boolean esCasoBase();
056        /**
057         * @return Las alternativas admisibles para el problema asociado
058         */
059        List<A> getAlternativas();
060        /**
061         * <p> Para poder calcular la solución del problema inicial en el estado final es muy recomendable que el estado disponga de una propiedad de tipo List&lt;A&gt; 
062         * que almacena se secuencia de alternativas tomadas. </p>.
063         * 
064         * @pre esCasoBase()
065         * @return Una solución del problema inicial y null si no hay solución para el problema inicial tras la secuencia de alternativas tomadas
066         * @throws IllegalStateException - Si se invoca el método en un estado no final
067         */
068        S getSolucion();
069        
070        /**
071         * @pre El método sólo es relevante si se usa la técnica con filtro
072         * @pos 
073         * <ul>
074         * <li> Si el problema es de minimización el valor debe ser 
075         * una cota inferior del valor de la propiedad objetivo del problema inicial, 
076         * asumiendo que se ha alcanzado el problema actual y se  va a seguir la alternativa <code> a </code>.
077         * La alternativa puede ser descartada si <code> getObjetivoEstimado(a) &gt; =  AlgoritmoPD.getMejorValor() </code>,
078         * puesto que el valor objetivo de la solución obtnida estaría por encima del mejor valor obtenido hasta el momento.
079         * Es decir nos quedamos con las alternativas que cumplen 
080         * <code> getObjetivoEstimado(a) &lt;   AlgoritmoPD.getMejorValor() </code>.
081         * Donde <code> AlgoritmoPD.getMejorValor() </code> es el mejor valor encontrado por el algoritmo hasta ahora.
082         * <li> Si el problema original fuera de maximización el valor debe ser 
083         * una cota superior del valor de la propiedad objetivo del problema inicial, 
084         * asumiendo que se ha alcanzado el problema actual y se  va a seguir la alternativa <code> a </code>.
085         *  Es decir nos quedamos con las alternativas que cumplen 
086         * <code> getObjetivoEstimado(a) &gt;   AlgoritmoPD.getMejorValor() </code>.
087         * </ul>
088         * @param a - Alternativa elegida
089         * @return El valor objetivo estimado si seguimos <code> a </code>
090         */
091        default Double getObjetivoEstimado(A a){
092                return null;
093        }
094        
095}
