050         * @param a Un entero
051         * @param b Un entero
052         * @param c Un entero
053         * @return Un stream de enteros que forman una progresión 
054         * aritmética desde a hasta b con razón c sin incluir el valor b.
055         */
056        public static IntStream range(Integer a, Integer b, Integer c){
057                Preconditions.checkArgument(a==b ||(b-a)*c > 0);
058                Integer hasta = (b-a)/c;
059                if((b-a)%c == 0){
060                        hasta = hasta - 1;
061                } 
062                return IntStream.rangeClosed(0, hasta).map(x->a+c*x);
063        }

106         * @param st Un stream
107         * @return Un elemento del stream escogido aleatoriamente
108         */
109        public static <T> T elementRandom(Stream<T> st){
110                List<T> ls = st.collect(Collectors.toList());
111                Integer n = ls.size();
112                return ls.get(Math2.getEnteroAleatorio(0, n));

115        /**
116         * @param <T> Tipo de los elementos del primer stream
117         * @param <U> Tipo de los elementos del segundo stream
118         * @param <R> Tipo de los elementos del stream resultado
119         * @param s1 Un Stream
120         * @param s2 Un Segundo Stream
121         * @param f Una Bifunction que opera un elemento del primer stream con un del segundo para 
122         * dar un resultado
123         * @return El resultado de operar los pares posibles de s1 y s2 con la bifunción f
124         */
125        public static <T, U, R> Stream<R> cartesianProduct(Stream<T> s1, Stream<U> s2, BiFunction<T, U, R> f) {
126                List<U> ls = s2.collect(Collectors.toList());
127                return s1.flatMap(x->ls.stream().map(y->f.apply(x,y)));
128        }

140        public static <T, U, R> List<R> cartesianProduct(Collection<T> s1, Collection<U> s2, BiFunction<T, U, R> f) {
141                return s1.stream().<R>flatMap(x->s2.stream().map(y->f.apply(x,y))).collect(Collectors.toList());

149                try {

164        /**
165         * @param v0 Un valor inicial
166         * @param f Una función para calcular el elemento siguiente
167         * @param pt Un predicado para comprobar que cada valor está en un dominio
168         * @param <T> El tipo de los elementos de la secuencia
169         * @return Un stream formados por valores que comienzan en v0, continuan por los siguientes valores 
170         * calculados por f mientras que se cumpla p
171         */
172        public static <T> Stream<T> whilePredicate(T v0, UnaryOperator<T> f, Predicate<T> pt) {
173                Stream.Builder<T> b = Stream.builder();
174                Stream.iterate(v0, f).peek(x->b.accept(x)).filter(pt.negate()).findFirst();
175                return b.build().filter(pt);
176        }

182         * @param pt Un predicado

185        public static <T> Stream<T> whilePredicate(Stream<T> st, Predicate<T> pt) {

256        /**
257         * @param sm Un String
258         * @param <E> El tipo de los elementos de la secuencia
259         * @return Un stream formado por los pares de elementos formados por un elemento y el siguiente en sm
260         */
261        public static <E> Stream<Tuple2<E,E>> toPairStream(Stream<E> sm) {
262                List<E> ls = new ArrayList<>();
263                ls.add(null);
264                Stream<Tuple2<E,E>> r = sm
265                                        .map(e->Tuple2.create(ls.get(0), e))    
266                                        .peek(p-> ls.set(0,p.getV2()))
267                                        .filter(p-> p.getV1()!=null);
268                return r;
269        }

282        /**
283         * @param stream1 Un stream
284         * @param stream2 Un segundo stream
285         * @param f1 Una función que calcula una clave para los elementos de stream1
286         * @param f2 Una función que calcula una clave para los elementos de stream2
287         * @param fr Una función que calcula un nuevo valor a partir  de uno procedente de stream1 y otro del stream2
288         * @param <T> El tipo de los elementos de la primera secuencia
289         * @param <U> El tipo de los elementos de la segunda secuencia
290         * @param <K> El tipo de los elementos de la clave
291         * @param <R> El tipo de los elementos de la secuencia resultante
292         * @return Un stream resultado del joint de stream1 y stream2
293         */
294        public static <T, U, K, R> Stream<R> joint(Stream<T> stream1,
295                        Stream<U> stream2,
296                        Function<? super T, ? extends K> f1,
297                        Function<? super U, ? extends K> f2, 
298                        BiFunction<T, U, R> fr) {
299
300                final Map<K, List<T>> map1 = stream1.collect(Collectors.groupingBy(f1));
301                final Map<K, List<U>> map2 = stream2.collect(Collectors.groupingBy(f2));
302
303                Set<K> sk = new HashSet<K>(map1.keySet());
304                sk.retainAll(map2.keySet());
305                Stream<R> r = sk.stream()
306                                .<R>flatMap(x -> Streams.cartesianProduct(map1.get(x), map2.get(x), fr).stream());
307                return r;
308        }
