122         * @pre el problema es de minimización o de maximización
123         * @param p Un problema
124         * @return Un árbol con las alternativas que están dentro de la solución
125         */
126        public Tree<A> getAlternativasDeSolucion(ProblemaPD<S,A> p){
127                Tree<A> r = null;
128                if (!p.esCasoBase()) {
129                        Sp<A> sp = this.getSolucionParcial(p);
130                        List<Tree<A>> la = Lists.newArrayList();
131                        Integer np = p.getNumeroSubProblemas(sp.alternativa);
132                        for (Integer i = 0; i < np; i++) {
133                                Tree<A> rh = this.getAlternativasDeSolucion(p.getSubProblema(sp.alternativa, i));
134                                if (rh!=null) {
135                                        la.add(rh);
136                                }
137                        }
138                        r = Tree.create(sp.alternativa,la);
139                }
140                return r;
141        }

143        public AlgoritmoPD(Iterable<ProblemaPD<S,A>> ps){               
144            problema = Iterables.get(ps, 0);
145            problemas = ps;
146            tipo = problema.getTipo();
147            this.numeroDeProblemas = 0;
148            this.idsDeProblemas = HashBiMap.create();
149            this.problemasDesdeId = this.idsDeProblemas.inverse();
150            mejorValor = isMin()? Double.MAX_VALUE: Double.MIN_VALUE;  
151        }
152        
153        public void ejecuta() {
154                if (AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.setTiempoDeEjecucionInicial();
155                solucionesParciales = Maps.newHashMap();
156                for (ProblemaPD<S, A> p : problemas) {
157                        do {
158                                pD(p);
159                        } while (isRandomize && solucionesParciales.get(problema) == null);
160                }
161                if (AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.setTiempoDeEjecucionFinal();
162        }

164        /**
165         * @param <S> El tipo de la solución
166         * @param <A> El tipo de la alternativa
167         * @param p Un problema
168         * @param alternativas Sus alternativas
169         * @return Una alternativa escogida al azar o todas ellas dependiendo del tamaño del problam
170         */
171        private List<A> randomize(ProblemaPD<S,A> p, List<A> alternativas){
172                List<A> alt;
173                if(isRandomize && p.size()>sizeRef){
174                        List<A> ls = Lists.newArrayList(alternativas);                  
175                        alt = Lists2.randomUnitary(ls);
176                }else{
177                        alt = alternativas;
178                }
179                return alt;
180        }

182        private void actualizaMejorValor(ProblemaPD<S,A> p){
183                        Double objetivo = p.getObjetivo();
184                        if (objetivo!=null &&
185                                        (isMin() && objetivo < mejorValor || 
186                                         isMax()&& objetivo > mejorValor)) {
187                                if (AlgoritmoPD.metricasOK)
188                                        AlgoritmoPD.metricas.addActualizacionMejorValor();
189                                mejorValor = objetivo;
190                        }
191        }

193        protected void guardaEnMemoria(ProblemaPD<S, A> p, Sp<A> e) {
194                this.numeroDeProblemas++;
195                this.idsDeProblemas.put(p,numeroDeProblemas);
196                solucionesParciales.put(p, e);
197        }

199        /**
200         * @param <S> El tipo de la solución
201         * @param <A> El tipo de la alternativa
202         * @param p El problema a resolver 
203         * @return Su solución parcial
204         */
205        private Sp<A> pD(ProblemaPD<S,A> p){
206                if(AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.addLLamadaRecursiva();          
207                Sp<A> e = null; 
208                if (haSidoResueltoYa(p)){
209                        if(AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.addUsoDeLaMemoria();
210                        e = getSolucionParcial(p);
211                } else if(p.esCasoBase()) {
212                        if(AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.addCasoBase();
213                        e = p.getSolucionParcialCasoBase();
214                        guardaEnMemoria(p, e); 
215                } else if(p.estaFueraDeRango()){
216                        if(AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.addFueraDeRango();
217                } else {
218                        List<Sp<A>> solucionesDeAlternativas = Lists.newArrayList(); 
219                        for(A a: randomize(p,p.getAlternativas())){
220                                if(conFiltro && isMin() && p.getObjetivoEstimado(a) >= mejorValor) {
221                                        if(AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.addUnFiltro();
222                                        continue;
223                                }
224                                if(conFiltro && isMax() && p.getObjetivoEstimado(a) <= mejorValor) {
225                                        if(AlgoritmoPD.metricasOK) AlgoritmoPD.metricas.addUnFiltro();
226                                        continue;
227                                }
228                                int numeroDeSubProblemas = p.getNumeroSubProblemas(a);
229                                List<Sp<A>> solucionesDeSubProblemas = Lists.newArrayList();  
230                                boolean haySolucion = true;
231                                for(int i = 0; i < numeroDeSubProblemas; i++){
232                                        ProblemaPD<S,A> pr = p.getSubProblema(a,i); 
233                                        Sp<A> sp = pD(pr);
234                                        if(sp==null) { haySolucion=false; break;}                                       
235                                        solucionesDeSubProblemas.add(sp);                       
236                                }
237                                Sp<A> sa = haySolucion?p.getSolucionParcialPorAlternativa(a, solucionesDeSubProblemas): null;
238                                solucionesDeAlternativas.add(sa);
239                        }
240                        solucionesDeAlternativas = solucionesDeAlternativas.stream().
241                                        filter(x -> x != null).collect(Collectors.toList());
242                        if (!solucionesDeAlternativas.isEmpty()) {
243                                e = p.getSolucionParcial(solucionesDeAlternativas);
244                        }
245                        if(e!=null) {
246                                e.solucionesDeAlternativas = solucionesDeAlternativas;
247                        }
248                        guardaEnMemoria(p, e);                  
249                }
250                if (conFiltro && solucionesParciales.get(p)!=null) {
251                        actualizaMejorValor(p);
252                }
253                return e;
254        }

257        /**
258         * @param pd - Problema del que se quiere obtener la solución
259         * @return Solución del problema o null si no tiene solución o no ha ha sido encontrado por el algoritmo
260         */
261        @SuppressWarnings("unchecked")
262        public S getSolucion(ProblemaPD<S, A> pd) {     
263                S s = null;
264                if (solucionesParciales.containsKey(pd)) {
265                        Sp<A> e = solucionesParciales.get(pd);
266                        if (e != null) {
267                                if (pd.esCasoBase()) {                                  
268                                        s = pd.getSolucionReconstruidaCasoBase(e);
269                                } else if (e.alternativa != null) {
270                                        List<S> soluciones = Lists.<S> newArrayList();
271                                        for (int i = 0; i < pd.getNumeroSubProblemas(e.alternativa); i++) {
272                                                soluciones.add(getSolucion(pd.getSubProblema(e.alternativa, i)));
273                                        }
274                                        s = pd.getSolucionReconstruidaCasoRecursivo(e, soluciones);
275                                } else if (e.alternativa == null) {
276                                        s = (S) e.propiedad;
277                                }
278                        }
279                }
280                return s;
281        }

301        private void marcarEnSolucion(ProblemaPD<S,A> pd){
302                if(solucionesParciales.containsKey(pd)){
303                        Sp<A> e = solucionesParciales.get(pd);          
304                        if(e!=null){
305                                e.estaEnLaSolucion =true;
306                                A alternativa = e.alternativa;                  
307                                if (!pd.esCasoBase()) {
308                                        for (int i = 0; i < pd.getNumeroSubProblemas(alternativa); i++) {
309                                                ProblemaPD<S, A> pds = pd.getSubProblema(alternativa, i);
310                                                marcarEnSolucion(pds);
311                                        }
312                                }       
313                        }
314                }
315        }

317        private String problema(ProblemaPD<S,A> p, Sp<A> e){
318                String s= "    "+"\""+this.idsDeProblemas.get(p)+"\"";
319                if(e!=null){
320                        if (p.esCasoBase()) {
321                                s = s + " [shape=box, style=dotted, label=\"" + p + "\"]";
322                        }else{
323                                s = s + " [shape=box, label=\"" + p + "\"]";
324                        }
325                } else{
326                        s = s+" [shape=diamond, label=\""+p+"\"]";
327                }
328                return s+";";
329        }

331        private String alternativa(ProblemaPD<S,A> p, A alternativa){
332                String s = "    "+"\""+this.idsDeProblemas.get(p)+","+alternativa+"\""+" [label="+alternativa+"]";
333                return s+";";
334        }

336        private String aristaProblemaToAlternativa(ProblemaPD<S,A> p, A alternativa, Sp<A> e){
337                String s = "    "+"\""+this.idsDeProblemas.get(p)+"\""+" -> "+"\""+this.idsDeProblemas.get(p)+","+alternativa+"\"";
338                if(e.estaEnLaSolucion && e.alternativa.equals(alternativa)){
339                        s = s+ "[style=bold,arrowhead=dot]";
340                }
341                return s+";";
342        }

344        private String aristaAlternativaToProblema(ProblemaPD<S,A> p, A alternativa, ProblemaPD<S,A> ps, Sp<A> e){
345                String s = "    "+"\""+this.idsDeProblemas.get(p)+","+alternativa+"\""+" -> "+"\""+this.idsDeProblemas.get(ps)+"\"";
346                if(e.estaEnLaSolucion && e.alternativa.equals(alternativa)){
347                        s = s+ "[style=bold,arrowhead=dot]";
348                }
349                return s+";";

353        private void showAll(ProblemaPD<S, A> p, Set<ProblemaPD<S,A>> visitados) {
354                if(visitados.contains(p)) return;
355                visitados.add(p);
356                Sp<A> e = solucionesParciales.get(p);
357                super.getFile().println(problema(p, e));
358                if (!p.esCasoBase()) {
359                                for (Sp<A> solParAlt : e.solucionesDeAlternativas) {
360                                        super.getFile().println(alternativa(p, solParAlt.alternativa));
361                                        super.getFile().println(aristaProblemaToAlternativa(p,solParAlt.alternativa, e));
362                                        for (int i = 0; i < p.getNumeroSubProblemas(solParAlt.alternativa); i++) {
363                                                ProblemaPD<S, A> pds = p.getSubProblema(solParAlt.alternativa, i);
364                                                super.getFile().println(aristaAlternativaToProblema(p,solParAlt.alternativa, pds, e));
365                                                showAll(pds,visitados);
366                                        }
367                                }
368                }
369        }

371        /**
372         * Un tipo diseñado para representar soluciones parciales a partir de las cuales 
373         * se puede reconstruir la solución del problema. 
374         * Esta formado por un par: una alternativa y el valor de una propiedad. La solución del problema 
375         * es la que se obtendría tomando la alternativa y el valor estaría en la propiedad
376         * 
377         * El valor null para este tipo representará la no existencia de solución
378         * 
379         * @param <A1> Tipo de la alternativa
380         */
381        public static class Sp<A1> implements Comparable<Sp<A1>> {
382                
383
384                public A1 alternativa;
385                public Double propiedad;
386                private List<Sp<A1>> solucionesDeAlternativas = null; 
387                private boolean estaEnLaSolucion = false;
388                                        
389                public static <A2> Sp<A2> create(A2 alternativa,Double propiedad){
390                        return new Sp<A2>(alternativa, propiedad);
391                }
392
393                protected Sp(A1 alternativa, Double propiedad) {
394                        super();
395                        if(propiedad==null) throw new IllegalArgumentException("Propiedad es null");
396                        this.alternativa = alternativa;
397                        this.propiedad = propiedad;     
398                }               
399                
400                @Override
401                public String toString(){
402                        return "("+alternativa+","+propiedad+")";
403                }
404
405                @Override
406                public int compareTo(Sp<A1> ob) {
407                        return this.propiedad.compareTo(ob.propiedad);
408                }
409                
410        }
