001package us.lsi.pd;
002import java.util.*;
003
004import us.lsi.pd.AlgoritmoPD.Sp;
005
006
007/**
008 * <p>Interface que debe implementar un problema para ser resuelto por la técnica de Programación Dinámica </p>
009 * 
010 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema14.pdf" target="_blank">Tema14</a></p>
011 * 
012 * @author Miguel Toro
013 *
014 * @param <S> El tipo de la solución del problema
015 * @param <A> El tipo de las alternativas
016 * @see AlgoritmoPD.Sp
017 */
018public interface ProblemaPD<S,A>  {     
019        
020        public enum Tipo{Min,Max,Otro}
021        
022        
023        /**
024         * @return El tipo del problema: minimiza, maximiza o busca un valor que involucra a todas las alternativas
025         *
026         * 
027         **/
028        Tipo getTipo(); 
029        /**
030         * @return El tamaño del problema
031         */
032        int size();
033        
034        /**
035         * @return Si el problema está fuera de rango y su solución es null
036         */
037        default boolean estaFueraDeRango(){
038                return false;
039        }
040        /**
041         * @return Si el problema es un caso base
042         * 
043         */
044        boolean esCasoBase();
045        /**
046         * @pos 
047         * <ul>
048         *      <li> Si el problema no tiene solución se debe devolver null
049         *      <li> Si el problema tiene solución asociada a una alternativa esta será de la forma (a,p).Siendo a la alternativa y p la propiedad correspondiente.
050         *      <li> Si el problema tiene solución pero no asociada a una alternativa esta será de la forma (null,p).Siendo p la propiedad correspondiente.
051         *      </ul> 
052         * @return  La solución del caso base
053         */
054        Sp<A> getSolucionParcialCasoBase();
055        /**
056         * @pre ls!=null, ls no está vacía, ls no contiene null, ls no contiene valores de la forma <code> (a,null) </code>
057         * 
058         * @pos 
059         * <ul>
060         *  <li> Si ls es vacía tras eliminar los valores <code> (a,null) </code> la solución devuelta es null. El problema no tiene solución.
061         *      <li> Si el problema tiene solución asociada a una alternativa, esta será de la forma <code>(a,p)</code>. Siendo <code> a </code>la alternativa y <code> p </code> la propiedad correspondiente. 
062         *      <li> Si el problema tiene solución pero no asociada a una alternativa esta será de la forma <code> (null,p)</code>. Siendo <code> p </code> la propiedad correspondiente.
063         *      </ul> 
064         * @param ls - Soluciones existentes alcanzados tras tomar las diferentes alternativas
065         * @return La solución del problema
066         */
067        Sp<A> getSolucionParcial(List<Sp<A>> ls);
068        /**
069         * @pre 0 &lt; = i &lt; getNumeroSubProblemas(a)
070         * @param a - Alternativa escogida
071         * @param np - Número del subproblema
072         * @return Subproblema
073         */
074        ProblemaPD<S,A> getSubProblema(A a, int np);
075        /**
076         * @pre <code> ls.size() == getNumeroSubProblemas(a) </code>, a está incluida en <code> getAlternativas() </code> y
077         * ls no contiene null
078         * @param a - Alternativa escogida
079         * @param ls - Soluciones de los subproblemas
080         * @return La solución del problema alcanzado siguiendo la alternativa <code> a </code> 
081         * y las soluciones parciales de los subproblemas del mismo están en ls 
082         */
083        Sp<A> getSolucionParcialPorAlternativa(A a , List<Sp<A>> ls);   
084        /**
085         * @pos Si un problema no tiene solución el conjunto de alternativas es vacío 
086         * @return Las alternativas disponibles para el problema
087         */
088        List<A> getAlternativas();
089        /**
090         * @param a - Alternativa escogida
091         * @return Número de subproblemas para una alternativa dada
092         */
093        int getNumeroSubProblemas(A a);
094        /**
095         * @pre El problema es un caso base
096         * @pos 
097         * <ul>
098         *  <li> Si sp es de la forma <code> (a,p) </code> la solución se reconstruye a partir de esos dos valores. 
099         *  <li> Si es de la forma <code> (null,p) </code> sólo a partir de <code> p </code>.
100         * </ul> 
101         * @param sp - Solución parcial del caso base
102         * @return La solución para un caso base si la solución parcial es sp. 
103         */
104        S getSolucionReconstruidaCasoBase(Sp<A> sp);    
105        /**
106         * @pre 
107         * <p> El problema no es un caso base, <code> ls !=null </code>, <code> ls.size() &gt; 0 </code> y ls no contiene null.</p>
108         * <p> Si el tipo es Min o Max entonces sp = (a,p) </p>
109         * <p> Si tipo es Otro entonces sp es (null,p) y S debe ser Double. </p>
110         * @pos
111         * <ul>
112         *  <li> Si <code> sp = (a,p) </code> la solución se reconstruye a partir de <code> a, p, ls </code> estando en <code> ls </code> las 
113         *  soluciones de los subproblemas que se alcanzan siguiendo la alternativa <code> sp.alternativa </code> 
114         *  <li> Si <code> sp = (null,p) </code> la solución es igual a <code> sp.propiedad </code>.
115         * </ul> 
116         * @param sp - Solución parcial del problema
117         * @param ls - 
118         * <ul>
119         *  <li> Si <code> sp = (a,p)  </code> La solución de los subproblemas que se alcanzan 
120         *  siguiendo la alternativa <code> sp.alternativa </code>. 
121         *  <li> Si <code> sp = (null,p) </code> entonces <code> ls </code> es irrelevante. 
122         * </ul>
123         * @return La solución para el problema. 
124         */
125        S getSolucionReconstruidaCasoRecursivo(Sp<A> sp, List<S> ls);   
126        
127        /**
128         * @pre El uso del método es relevante si usamos la técnica con filtro. Si no la usamos el método es irrelevante.
129         * @pos 
130         * <ul>
131         * <li> Si el problema es de minimización el valor debe ser 
132         * una cota inferior del valor de la propiedad objetivo del problema inicial, 
133         * asumiendo que estamos en el problema actual y se  va a seguir la alternativa <code> a </code>.
134         * La alternativa puede ser descartada si <code> getObjetivoEstimado(a) &gt; =  AlgoritmoPD.getMejorValor() </code>,
135         * puesto que el valor objetivo de la solución obtenida estaría por encima del mejor valor obtenido hasta el momento.
136         * Es decir nos quedamos con las alternativas que cumplen 
137         * <code> getObjetivoEstimado(a)() &lt;   AlgoritmoPD.getMejorValor() </code>.
138         * Donde <code> AlgoritmoPD.getMejorValor() </code> es el mejor valor encontrado por el algoritmo hasta ahora.
139         * <li> Si el problema original fuera de maximización el valor debe ser 
140         * una cota superior del valor de la propiedad objetivo del problema actual, 
141         * asumiendo que estamos en el problema actual y se  va a seguir la alternativa <code> a </code>.
142         *  Es decir nos quedamos con las alternativas que cumplen 
143         * <code> getObjetivoEstimado(a) &gt;   AlgoritmoPD.getMejorValor() </code>.
144         * </ul>
145         * @param a - Alternativa elegida
146         * @return Una cota para valor objetivo si seguimos <code> a </code>
147         */
148        default Double getObjetivoEstimado(A a){
149                return null;
150        }
151        /**
152         * 
153         * @pre El uso del método es relevante si usamos la técnica con filtro. Si no la usamos el método es irrelevante.
154         * @return Valor de la propiedad objetivo del problema inicial si es posible calcularlo asumiendo
155         * que estamos en el problema actual. Si no es posible null.
156         */
157        default Double getObjetivo() {
158                return null;
159        }
160
