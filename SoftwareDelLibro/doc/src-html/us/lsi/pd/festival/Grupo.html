001package us.lsi.pd.festival;
002
003public class Grupo{
004
005        private Integer codigo;
006        private String nombre;
007        private Integer votos;
008        private Double precio;
009        private String dia;
010        private String hora;
011        private boolean cerca;
012        private boolean nuevo;
013        
014        private Grupo(Integer codigo, String nombre, Integer votos, Double precio,
015                        String dia, String hora, boolean cerca, boolean nuevo) {
016                super();
017                this.codigo = codigo;
018                this.nombre = nombre;
019                this.votos = votos;
020                this.precio = precio;
021                this.dia = dia;
022                this.hora = hora;
023                this.cerca = cerca;
024                this.nuevo = nuevo;
025        }
026        
027        private Grupo(int codigo, String[] fm) {
028                super();
029                this.codigo = codigo;
030                this.nombre = fm[0];
031                this.votos = new Integer(fm[1]);
032                this.precio = new Double(fm[2]);
033                this.dia = fm[3];
034                this.hora = fm[4];
035                this.cerca = new Boolean(fm[5]);
036                this.nuevo = new Boolean(fm[6]);
037        }
038        
039        public static Grupo create(Integer codigo, String nombre, Integer votos, Double precio,
040                        String dia, String hora, boolean cerca, boolean nuevo) {
041                return new Grupo(codigo, nombre, votos, precio, dia, hora, cerca, nuevo);
042        }
043        
044        public static Grupo create(int codigo, String[] fm) {
045                return new Grupo(codigo, fm);
046        }
047        
048        public Integer getCodigo() {
049                return codigo;
050        }
051        
052        public String getNombre() {
053                return nombre;
054        }
055        
056        public Integer getVotos() {
057                return votos;
058        }
059        
060        public Double getPrecio() {
061                return precio;
062        }
063        
064        public String getDia() {
065                return dia;
066        }
067        
068        public String getHora() {
069                return hora;
070        }
071
072        public boolean esCerca() {
073                return cerca;
074        }
075        
076        public boolean esNuevo() {
077                return nuevo;
078        }
079        
080        @Override
081        public int hashCode() {
082                final int prime = 31;
083                int result = 1;
084                result = prime * result + ((codigo == null) ? 0 : codigo.hashCode());
085                return result;
086        }
087        
088        @Override
089        public boolean equals(Object obj) {
090                if (this == obj)
091                        return true;
092                if (obj == null)
093                        return false;
094                if (getClass() != obj.getClass())
095                        return false;
096                Grupo other = (Grupo) obj;
097                if (codigo == null) {
098                        if (other.codigo != null)
099                                return false;
100                } else if (!codigo.equals(other.codigo))
101                        return false;
102                return true;
103        }
104
105        @Override
106        public String toString() {
107                return nombre;
108        }
109
110}
