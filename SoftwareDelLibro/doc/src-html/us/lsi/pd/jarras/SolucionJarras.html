001package us.lsi.pd.jarras;
002
003import java.util.List;
004import java.util.stream.Collectors;
005
006import us.lsi.bt.SolucionBT;
007
008import com.google.common.collect.Lists;
009
010public class SolucionJarras implements SolucionBT {
011
012        public static SolucionJarras create() {
013                return new SolucionJarras();
014        }
015
016        public static SolucionJarras create(List<Accion> m) {
017                return new SolucionJarras(m);
018        }
019
020        private List<Accion> listaAcciones;
021
022        private SolucionJarras(List<Accion> ls) {
023                super();
024                this.listaAcciones = Lists.newArrayList(ls);
025        }
026
027        private SolucionJarras() {
028                super();
029                this.listaAcciones = Lists.newArrayList();
030        }
031
032        public void add(Accion op) {
033                listaAcciones.add(op);
034        }
035
036        public void addFirst(Accion op) {
037                listaAcciones.add(0,op);
038        }
039        
040        public Accion removeLast() {
041                if (listaAcciones.isEmpty())
042                        return null;
043                return listaAcciones.remove(listaAcciones.size() - 1);
044        }
045
046        public Integer getNumAcc() {
047                return listaAcciones.size();
048        }
049
050        public List<Accion> getListaAcciones() {
051                return listaAcciones;
052        }
053
054        public void setListaOperaciones(List<Accion> ls) {
055                this.listaAcciones = ls;
056        }       
057
058        @Override
059        public Double getObjetivo() {
060                return (double) this.getNumAcc();
061        }
062        
063        @Override
064        public int hashCode() {
065                final int prime = 31;
066                int result = 1;
067                result = prime * result
068                                + ((listaAcciones == null) ? 0 : listaAcciones.hashCode());
069                return result;
070        }
071
072        @Override
073        public boolean equals(Object obj) {
074                if (this == obj)
075                        return true;
076                if (obj == null)
077                        return false;
078                if (!(obj instanceof SolucionJarras))
079                        return false;
080                SolucionJarras other = (SolucionJarras) obj;
081                if (listaAcciones == null) {
082                        if (other.listaAcciones != null)
083                                return false;
084                } else if (!listaAcciones.equals(other.listaAcciones))
085                        return false;
086                return true;
087        }
088
089        @Override
090        public String toString() {
091                return "SolucionJarra [numOp=" + this.getNumAcc() + "\n" + 
092        listaAcciones.stream().map(x->x.toString()).collect(Collectors.joining("\n")) + "]";
093        }       
094
095}
