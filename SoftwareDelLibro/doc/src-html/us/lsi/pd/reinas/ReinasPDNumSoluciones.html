001package us.lsi.pd.reinas;
002
003import java.util.ArrayList;
004import java.util.HashSet;
005import java.util.List;
006import java.util.Set;
007import java.util.stream.Collectors;
008import java.util.stream.IntStream;
009import java.util.stream.Stream;
010
011import us.lsi.pd.AlgoritmoPD.Sp;
012import us.lsi.pd.ProblemaPD;
013import us.lsi.pd.ProblemaPDR;
014
015
016public class ReinasPDNumSoluciones implements ProblemaPDR<Integer, Integer> {
017
018        public static int numeroDeReinas = 8;
019        
020        public static ReinasPDNumSoluciones create(Integer x, List<Integer> yOcupadas) {
021                return new ReinasPDNumSoluciones(x, yOcupadas);
022        }
023
024        public static ReinasPDNumSoluciones create() {
025                return new ReinasPDNumSoluciones(0, new ArrayList<>());
026        }
027        
028        private Integer x;
029        private List<Integer> yOcupadas;
030        private Set<Integer> diagonalesPrincipalesOcupadas;
031        private Set<Integer> diagonalesSecundariasOcupadas;
032        
033        private void calculaPropiedadesDerivadas(){
034                this.diagonalesPrincipalesOcupadas = new HashSet<>();
035                this.diagonalesSecundariasOcupadas = new HashSet<>();
036                for(int x=0; x < this.yOcupadas.size();x++){
037                        this.diagonalesPrincipalesOcupadas.add(this.yOcupadas.get(x)-x);
038                        this.diagonalesSecundariasOcupadas.add(this.yOcupadas.get(x)+x);
039                }
040        }
041        
042        private ReinasPDNumSoluciones(Integer x, List<Integer> yOcupadas) {
043                super();
044                this.x= x;
045                this.yOcupadas = new ArrayList<Integer>(yOcupadas);
046                calculaPropiedadesDerivadas();
047        }
048
049        @Override
050        public us.lsi.pd.ProblemaPD.Tipo getTipo() {
051                return Tipo.Otro;
052        }
053        
054        @Override
055        public int size() {
056                return numeroDeReinas-this.x;
057        }
058
059        @Override
060        public boolean esCasoBase() {
061                return this.size() == 1;
062        }
063
064        @Override
065        public Sp<Integer> getSolucionParcialCasoBase() {
066                Integer n = this.getAlternativas().size();
067                return Sp.create(null, (double) n);
068        }
069
070        @Override
071        public Sp<Integer> getSolucionParcial(List<Sp<Integer>> ls) {
072                Double r = ls.stream().mapToDouble(sp->sp.propiedad).sum();
073                return Sp.create(null, r);
074        }
075
076        @Override
077        public ProblemaPD<Integer, Integer> getSubProblema(Integer a) {
078                List<Integer> ls = new ArrayList<>(this.yOcupadas);
079                ls.add(a);
080                return ReinasPDNumSoluciones.create(this.x+1,ls);
081        }
082
083        @Override
084        public Sp<Integer> getSolucionParcialPorAlternativa(Integer a, Sp<Integer> s) {
085                return Sp.create(a, s.propiedad);
086        }
087        
088        @Override
089        public List<Integer> getAlternativas() {                
090                Stream<Integer> s = IntStream
091                                .range(0, ReinasPDNumSoluciones.numeroDeReinas)
092                                .filter(y -> !this.yOcupadas.contains(y)
093                                                && !this.diagonalesPrincipalesOcupadas.contains(y-this.x)
094                                                && !this.diagonalesSecundariasOcupadas.contains(y+this.x)
095                                                )
096                                .boxed();
097                return s.collect(Collectors.toList());
098        }
099
100        @Override
101        public Integer getSolucionReconstruidaCasoBase(Sp<Integer> sp) {
102                return sp.propiedad.intValue();
103        }
104
105        @Override
106        public Integer getSolucionReconstruidaCasoRecursivo(Sp<Integer> sp, Integer s) {
107                return sp.propiedad.intValue();
108        }       
109        
110}
