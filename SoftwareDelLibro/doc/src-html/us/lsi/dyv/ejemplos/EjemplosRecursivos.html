003public class EjemplosRecursivos {
004
005        public static long binomio(long n, long k){
006                long r;
007                if(k == 0 || k == n){
008                        r = 1;
009                }else if (k == 1 || k == n-1){
010                        r = n;
011                }else {
012                        r = binomio(n-1,k-1)+binomio(n-1,k);
013                }
014                return r;
015        }
016        
017        public static long fib(int n){
018                long r;
019                if(n<=1){
020                        r = n;
021                }else {
022                        r = fib(n-1)+fib(n-2);          
023            }
024                return r;
025        }
026        
027        public static void main(String[] args) {
028                System.out.println(binomio(10L,5L));
029        }
030
031}
